001package eu.andredick.tests;
002
003import eu.andredick.aco.algorithm.AbstractAlgorithm;
004import eu.andredick.configuration.AbstractAlgorithmConfiguration;
005import eu.andredick.configuration.AlgorithmConfiguration_Stigmergy;
006import eu.andredick.configuration.AlgorithmConfiguration_SubsetPairs;
007import eu.andredick.orlib.OrlibConverter;
008import eu.andredick.scp.SCProblem;
009import eu.andredick.tools.ChartTools;
010
011import java.io.File;
012
013public class Main {
014
015    public static void main(String[] args) {
016
017        // Holen einer Probleminstanz
018        SCProblem problem = OrlibConverter.getProblem("scpd1");
019
020        problem.isConsistent();
021
022        // Es können mehrere Konfigurationen des Algorithmus gegeneinander getestet werden
023        // Konfig A
024        AbstractAlgorithmConfiguration config1 = new AlgorithmConfiguration_Stigmergy();
025        AbstractAlgorithm abstractAlgorithm1 = config1.create(problem);
026        Thread thread1 = new Thread(abstractAlgorithm1);
027
028        // Konfig B
029        AbstractAlgorithmConfiguration config2 = new AlgorithmConfiguration_SubsetPairs();
030        AbstractAlgorithm abstractAlgorithm2 = config2.create(problem);
031        Thread thread2 = new Thread(abstractAlgorithm2);
032
033        // Erzeugen von Threads für parallele Ausführung
034        thread1.start();
035        thread2.start();
036
037
038        try {
039
040            // Warten bis Threads beendet sind
041            thread1.join();
042            thread2.join();
043
044            System.out.println("thread ende");
045
046            ChartTools chart = new ChartTools("Benchmark", "Interation", "z_best");
047
048            chart.addDataSeries(config1.getConfigName(), abstractAlgorithm1.getStatistics().getIterationMinValuesArray());
049            chart.addDataSeries(config2.getConfigName(), abstractAlgorithm2.getStatistics().getIterationMinValuesArray());
050
051            chart.saveChartAsSVG(new File("chart.svg"));
052
053        } catch (InterruptedException e) {
054            e.printStackTrace();
055        }
056
057
058    }
059}
