001package eu.andredick.scp;
002
003import eu.andredick.aco.problem.AbstractProblem;
004
005/*
006 * Die Klasse repräsentiert das Set Covering AbstractProblem.
007 */
008public class SCProblem extends AbstractProblem {
009
010    /**
011     * Name des konkreten Problems
012     */
013    protected String name;
014
015    /**
016     * Strukturmatrix [a_ij]
017     */
018    private Structure structure;
019
020    /**
021     * Zielfunktion
022     */
023    private ObjectiveFunction objectiveFunction;
024
025    /**
026     * Konstruktor
027     *
028     * @param relations                     Boolsche Matrix m x n
029     * @param objectiveFunctionCoefficients Koeffizienten der Zielfunktion c_j
030     */
031    public SCProblem(boolean[][] relations, float[] objectiveFunctionCoefficients) {
032        this.structure = new Structure(relations);
033        this.objectiveFunction = new ObjectiveFunction(objectiveFunctionCoefficients);
034        this.name = null;
035    }
036
037    /**
038     * Konstruktor
039     *
040     * @param relations                     Boolsche Matrix m x n
041     * @param objectiveFunctionCoefficients Koeffizienten der Zielfunktion c_j
042     * @param name                          Name des Problems
043     */
044    public SCProblem(boolean[][] relations, float[] objectiveFunctionCoefficients, String name) {
045        this.structure = new Structure(relations);
046        this.objectiveFunction = new ObjectiveFunction(objectiveFunctionCoefficients);
047        this.name = name;
048    }
049
050    /**
051     * Liefert den Namen des Problems
052     *
053     * @return Name des Problems
054     */
055    public String getName() {
056        return this.name;
057    }
058
059    /**
060     * Liefert die Strukturmatrix
061     *
062     * @return Strukturmatrix
063     */
064    public Structure getStructure() {
065        return this.structure;
066    }
067
068    /**
069     * Liefert die Zielfunktion
070     *
071     * @return Zielfunktion
072     */
073    public ObjectiveFunction getObjectiveFunction() {
074        return this.objectiveFunction;
075    }
076
077
078    /**
079     * Prüft auf Konsistenz des SCP-Problems
080     *
081     * @return Wahr, wenn konsistent
082     */
083    public boolean isConsistent() {
084        System.out.println("SCProblem isConsistent ...");
085        boolean[][] relations = this.getStructure().getRelations();
086        boolean success = true;
087        int sumCovers_i = 0;
088        for (int i = 0; i < relations.length; i++) {
089            int sum_i = 0;
090            for (int j = 0; j < relations[i].length; j++) {
091                if (relations[i][j]) sum_i++;
092            }
093            if (sum_i == 0) {
094                System.out.println(" Zeile " + i + " von keiner Spalte überdeckt");
095                success = false;
096            }
097            if (this.structure.getSubsetsWithElement(i).size() != sum_i) {
098                System.out.println(" Zeile " + i + " nicht konsistent");
099                success = false;
100            }
101            sumCovers_i += sum_i;
102        }
103        System.out.println(" Mindestens 1fache Überdeckung in Matrix - OK!");
104        int sumCovers = 0;
105        for (int i = 0; i < structure.elementsSize(); i++) {
106            int coveredTimes = structure.getSubsetsWithElement(i).size();
107            sumCovers += coveredTimes;
108            if (coveredTimes < 1) System.out.println(" Element " + i + " nicht überdeckt");
109        }
110        float a = sumCovers / structure.elementsSize();
111        float b = sumCovers_i / relations.length;
112        System.out.println(" Durchschnittliche Überdeckung: " + a + " " + b);
113
114        System.out.println("SCProblem isConsistent ..." + success);
115        return success;
116    }
117}
