001package eu.andredick.tools.orlib;
002
003import eu.andredick.scp.SCProblem;
004import eu.andredick.tools.ArrayTools;
005
006import java.io.BufferedReader;
007import java.io.IOException;
008import java.io.InputStream;
009import java.io.InputStreamReader;
010
011public class ConverterStandard implements AbstractConverter {
012
013    @Override
014    public SCProblem convert(InputStream in) throws IOException {
015        BufferedReader reader = new BufferedReader(new InputStreamReader(in));
016        String zeile = null;
017        zeile = reader.readLine();
018        int[] fileRow = ArrayTools.stringToIntArray(zeile);
019        int rows = fileRow[0];
020        int columns = fileRow[1];
021        boolean[][] ralations = ArrayTools.getZeroBool2DArray(rows, columns);
022        float[] objCoeff = new float[columns];
023
024        System.out.println("Rows: " + rows + " Cols:" + columns);
025
026        int coeffCounter = 0;
027        while (coeffCounter < objCoeff.length - 1) {
028            zeile = reader.readLine();
029            fileRow = ArrayTools.stringToIntArray(zeile);
030            for (int i = 0; i < fileRow.length; i++) {
031                objCoeff[i + coeffCounter] = fileRow[i];
032            }
033            coeffCounter += fileRow.length;
034        }
035        System.out.println("coeffCounter: " + coeffCounter);
036
037        for (int i = 0; i < rows; i++) {
038            zeile = reader.readLine();
039            fileRow = ArrayTools.stringToIntArray(zeile);
040            if (fileRow.length != 1) {
041                System.out.println("Mehr als eine Zahl enthalten");
042            }
043            int columnsInRow = fileRow[0];
044            int counter = 0;
045            while (counter < columnsInRow) {
046                zeile = reader.readLine();
047                fileRow = ArrayTools.stringToIntArray(zeile);
048                for (int j = 0; j < fileRow.length; j++) {
049                    int col = fileRow[j];
050                    ralations[i][col - 1] = true;
051                }
052                counter += fileRow.length;
053            }
054
055        }
056        return new SCProblem(ralations, objCoeff);
057    }
058}
