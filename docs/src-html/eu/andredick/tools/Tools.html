001package eu.andredick.tools;
002
003import java.util.ArrayList;
004import java.util.Random;
005
006public class Tools {
007
008    public static ArrayList<Integer> getIndexList(int size) {
009        ArrayList<Integer> indices = new ArrayList<Integer>();
010        for (int i = 0; i < size; i++) {
011            indices.add(i);
012        }
013        return indices;
014    }
015
016    /*All--------------------------------------*/
017    public static int[][] getRandomSCP(int m, int n) {
018        int[][] relations = new int[m][n];
019
020        Random r = new Random();
021
022        for (int i = 0; i < m; i++) {
023            int counter = 0;  //Sicherstellen, dass jedes Element in min Subset vorhanden
024            for (int j = 0; j < n; j++) {
025                relations[i][j] = r.nextBoolean() ? 1 : 0;
026                counter += relations[i][j];
027            }
028            if (counter == 0) {
029                relations[i][r.nextInt(n)] = 1;
030            }
031        }
032        return relations;
033    }
