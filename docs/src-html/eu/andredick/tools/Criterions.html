001package eu.andredick.tools;
002
003import eu.andredick.aco.algorithm.Statistics;
004
005import java.util.ArrayList;
006import java.util.List;
007
008public class Criterions {
009
010    public static float RAR(float z, float z_opt) {
011        return z / z_opt;
012    }
013
014    public static float RPZ(float z, float z_opt, float z_max) {
015        return (z - z_opt) / (z_max - z_opt) * 100;
016    }
017
018    public static float RPD(float z, float z_opt) {
019        return (z - z_opt) / (z_opt - 0) * 100;
020    }
021
022    public static float RPI(float z, float z_opt, float z_iter) {
023        return (z_iter - z) / (z_iter - z_opt) * 100;
024    }
025
026    public static float getAverageBestValue(Statistics[] statistics) {
027        float sum = 0;
028        for (Statistics s : statistics) {
029            sum += s.getGlobalMinValue();
030        }
031        return sum / statistics.length;
032    }
033
034    public static List<float[]> getArraysOfIterationBestValues(Statistics[] statistics) {
035        List<float[]> arrays = new ArrayList<>();
036        for (Statistics s : statistics) {
037            arrays.add(s.getIterationMinValuesArray());
038        }
039        return arrays;
040    }
041}
