001package eu.andredick.tools;
002
003import javax.swing.*;
004import javax.swing.text.Document;
005import javax.swing.text.html.HTMLEditorKit;
006import javax.swing.text.html.StyleSheet;
007import java.awt.*;
008
009/**
010 * A complete Java class that demonstrates how to create an HTML viewer with styles,
011 * using the JEditorPane, HTMLEditorKit, StyleSheet, and JFrame.
012 *
013 * @author alvin alexander, devdaily.com.
014 */
015public class HtmlEditorKitTest {
016    public HtmlEditorKitTest() {
017        SwingUtilities.invokeLater(new Runnable() {
018            public void run() {
019                // create jeditorpane
020                JEditorPane jEditorPane = new JEditorPane();
021
022                // make it read-only
023                jEditorPane.setEditable(false);
024
025                // create a scrollpane; modify its attributes as desired
026                JScrollPane scrollPane = new JScrollPane(jEditorPane);
027
028                // add an html editor kit
029                HTMLEditorKit kit = new HTMLEditorKit();
030                jEditorPane.setEditorKit(kit);
031
032                // add some styles to the html
033                StyleSheet styleSheet = kit.getStyleSheet();
034                styleSheet.addRule("body {color:#000; font-family:times; margin: 4px; }");
035                styleSheet.addRule("h1 {color: blue;}");
036                styleSheet.addRule("h2 {color: #ff0000;}");
037                styleSheet.addRule("pre {font : 10px monaco; color : black; background-color : #fafafa; }");
038
039                // create some simple html as a string
040                String htmlString = "<html>\n"
041                        + "<body>\n"
042                        + "<h1>Welcome!</h1>\n"
043                        + "<h2>This is an H2 header</h2>\n"
044                        + "<p>This is some sample text</p>\n"
045                        + "<p><a href=\"http://devdaily.com/blog/\">devdaily blog</a></p>\n"
046                        + "</body>\n";
047
048                // create a document, set it on the jeditorpane, then add the html
049                Document doc = kit.createDefaultDocument();
050                jEditorPane.setDocument(doc);
051                jEditorPane.setText(htmlString);
052
053                // now add it all to a frame
054                JFrame j = new JFrame("HtmlEditorKit Test");
055                j.getContentPane().add(scrollPane, BorderLayout.CENTER);
056
057                // make it easy to close the application
058                j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
059
060                // display the frame
061                j.setSize(new Dimension(300, 200));
062
063                // pack it, if you prefer
064                //j.pack();
065
066                // center the jframe, then make it visible
067                j.setLocationRelativeTo(null);
068                j.setVisible(true);
069            }
070        });
071    }
072
073    public static void main(String[] args) {
074        new HtmlEditorKitTest();
075    }
076}
