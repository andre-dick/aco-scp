009        boolean[][] array = new boolean[rows][cols];
010        for (int i = 0; i < rows; i++) {
011            for (int j = 0; j < cols; j++) {
012                array[i][j] = false;
013            }
014        }
015        return array;
016    }

029    public static ArrayList<Integer> getRandomIntPermutation(ArrayList<Integer> list) {
030        Random rGenerator = new Random();
031        int size = list.size();
032        for (int i = size - 1; i >= 1; i--) {
033            int j = rGenerator.nextInt(i + 1); // i+1 wegen exklusiver oberer Schranke
034            Integer t = list.get(i);
035            list.set(i, j);
036            list.set(j, t);
037        }
038        return list;
039    }

043    public static ArrayList<Integer> getRandomIntPermutation(int size) {
044        Random rGenerator = new Random();
045        ArrayList<Integer> list = new ArrayList<>();
046        for (int i = 0; i <= size - 1; i++) {
047            int j = rGenerator.nextInt(i + 1); // i+1 wegen exklusiver oberer Schranke
048            if (j != i) {
049                list.set(i, list.get(j));
050            }
051            list.set(j, i); // wobei i eigentlich eine Funktion f(i) ist
052        }
053        return list;
054    }

056    public static float[] getRandomFloatArray(int length, float from, float to) {
057        Random r = new Random();
058        float[] array = new float[length];
059        float delta = to - from;
060        for (int i = 0; i < array.length; i++) {
061            array[i] = from + r.nextFloat() * delta;
062        }
063
064        return array;
065    }

085    public static int[] stringToIntArray(String s) {
086        String[] items = s.trim().split("\\s+");
087        int[] results = new int[items.length];
088
089        for (int i = 0; i < items.length; i++) {
090            try {
091                results[i] = Integer.parseInt(items[i]);
092            } catch (NumberFormatException nfe) {
093                return null;
094            }
095            ;
096        }
097
098        return results;
099    }

101    public static float[] getAverageValuesOfArrays(List<float[]> arrays) {
102        int minLength = arrays.get(0).length;
103        int maxLength = arrays.get(0).length;
104        for (float[] a : arrays) {
105            if (a.length < minLength) {
106                minLength = a.length;
107            }
108            if (a.length > maxLength) {
109                maxLength = a.length;
110            }
111        }
112
113        float[] averageArray = new float[minLength];
114        for (int i = 0; i < minLength; i++) {
115            float sum = 0f;
116            for (float[] array : arrays) {
117                sum += array[i];
118            }
119            averageArray[i] = sum / (float) arrays.size();
120        }
121        return averageArray;
122    }

124    public static float[] getMaxValuesOfArrays(List<float[]> arrays) {
125        int minLength = arrays.get(0).length;
126        int maxLength = arrays.get(0).length;

128            if (a.length < minLength) {
129                minLength = a.length;
130            }
131            if (a.length > maxLength) {
132                maxLength = a.length;
133            }
134        }
135
136        float[] array = new float[minLength];
137        for (int i = 0; i < minLength; i++) {
138            float bestValue_i = arrays.get(0)[i];
139            for (int j = 1; j < arrays.size(); j++) {
140                if (arrays.get(j)[i] < bestValue_i) {
141                    bestValue_i = arrays.get(j)[i];
142                }
143            }
144            array[i] = bestValue_i;
145
146        }
147        return array;
148    }

208    public static float[] getProgressiveMinValueArray(float[] values) {
209        float[] progressiveMinValues = new float[values.length];
210        progressiveMinValues[0] = values[0];
211        for (int i = 1; i < values.length; i++) {
212            if (values[i] < progressiveMinValues[i - 1]) {
213                progressiveMinValues[i] = values[i];
214            } else {
215                progressiveMinValues[i] = progressiveMinValues[i - 1];
216            }
217        }
218        return progressiveMinValues;
219    }
