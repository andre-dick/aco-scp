001package eu.andredick.configuration;
002
003import eu.andredick.aco.algorithm.ACOAlgorithm;
004import eu.andredick.aco.ant.ACOAnt;
005import eu.andredick.aco.algorithm.AbstractAlgorithm;
006import eu.andredick.aco.ant.AbstractAnt;
007import eu.andredick.aco.combination.CombinationFactor;
008import eu.andredick.aco.construct.AbstractConstructionStrategy;
009import eu.andredick.aco.construct.ConstructionFromSubsets;
010import eu.andredick.aco.heuristic.HeuristicInfoSet;
011import eu.andredick.aco.heuristic.HeuristicRuleBestSubset;
012import eu.andredick.aco.heuristic.HeuristicRuleWeights;
013import eu.andredick.aco.localsearch.AbstractLocalSearchStrategy;
014import eu.andredick.aco.localsearch.LocalSearchStrategyNone;
015import eu.andredick.aco.masterprocess.AbstractMasterProcess;
016import eu.andredick.aco.masterprocess.MasterProcessBasic;
017import eu.andredick.aco.nextstep.AbstractNextStepStrategy;
018import eu.andredick.aco.nextstep.NextStepStrategyOnSubsetsDeterministic;
019import eu.andredick.aco.pheromoneassociation.PheromoneOnSubsets;
020import eu.andredick.aco.pheromoneevaporation.AbstractPheromoneEvaporation;
021import eu.andredick.aco.pheromoneevaporation.PheromoneEvaporation;
022import eu.andredick.aco.pheromoneinit.AbstractPheromoneInit;
023import eu.andredick.aco.pheromoneinit.PheromoneInit;
024import eu.andredick.aco.pheromoneperception.AbstractPheromonePerception;
025import eu.andredick.aco.pheromoneperception.PerceptionSimple;
026import eu.andredick.aco.pheromoneupdate.AbstractPheromoneUpdate;
027import eu.andredick.aco.pheromoneupdate.PheromoneUpdateOnSubsets;
028import eu.andredick.aco.solutionquality.SolutionQualityMin;
029import eu.andredick.aco.termination.TermCriterion;
030import eu.andredick.scp.SCPSolution;
031import eu.andredick.scp.SCProblem;
032
033public class AlgorithmConfiguration_Greedy extends AbstractAlgorithmConfiguration {
034
035
036    @Override
037    public void prepareConfigParameters() {
038
039        this.configName = "AlgorithmConfiguration_Greedy";
040
041        ConfigurationParameter<Float> phInitValue =
042                new ConfigurationParameter<>("pheromonInitValue", 1f);
043        this.addConfigurationParameter(phInitValue);
044
045        ConfigurationParameter<Float> evapFactor =
046                new ConfigurationParameter<>("evaporationFactor", 1f);
047        this.addConfigurationParameter(evapFactor);
048
049        ConfigurationParameter<Float> alpha =
050                new ConfigurationParameter<>("alpha", 0f);
051        this.addConfigurationParameter(alpha);
052
053        ConfigurationParameter<Float> beta =
054                new ConfigurationParameter<>("beta", 1f);
055        this.addConfigurationParameter(beta);
056
057        this.getParameter("antsize").setCurrentValue(1);
058
059    }
060
061    @Override
062    public AbstractAlgorithm create(SCProblem problem) {
063
064        PheromoneOnSubsets pheromoneStructure =
065                new PheromoneOnSubsets(problem);
066
067        float phInitValue = this.getParameter("pheromonInitValue").getCurrentValue().floatValue();
068        AbstractPheromoneInit pheromoneInitRule =
069                new PheromoneInit(phInitValue);
070
071        float evapFactor = this.getParameter("evaporationFactor").getCurrentValue().floatValue();
072        AbstractPheromoneEvaporation evaporationRule =
073                new PheromoneEvaporation(evapFactor);
074
075        pheromoneStructure.setEvaporationRule(evaporationRule);
076        pheromoneStructure.setPheromoneInitRule(pheromoneInitRule);
077
078        AbstractPheromoneUpdate updateRule =
079                new PheromoneUpdateOnSubsets(pheromoneStructure, new SolutionQualityMin());
080
081        float alpha = this.getParameter("alpha").getCurrentValue().floatValue();
082        float beta = this.getParameter("beta").getCurrentValue().floatValue();
083
084        HeuristicInfoSet heuristicInfoSet = new HeuristicInfoSet();
085        heuristicInfoSet.addRule(new HeuristicRuleWeights());
086        heuristicInfoSet.addRule(new HeuristicRuleBestSubset());
087
088        AbstractPheromonePerception perceptionRule = new PerceptionSimple();
089
090        AbstractNextStepStrategy nextStepRule =
091                new NextStepStrategyOnSubsetsDeterministic(
092                        pheromoneStructure, perceptionRule, heuristicInfoSet, new CombinationFactor(alpha, beta));
093
094        AbstractConstructionStrategy constructionStrategy =
095                new ConstructionFromSubsets(nextStepRule);
096
097        AbstractLocalSearchStrategy localSearchStrategy =
098                new LocalSearchStrategyNone();
099
100        int antSize = this.getParameter("antsize").getCurrentValue().intValue();
101
102        AbstractAnt[] ants = new AbstractAnt[antSize];
103        for (int i = 0; i < ants.length; i++) {
104            ants[i] = new ACOAnt<SCPSolution, SCProblem>(problem, updateRule, constructionStrategy, localSearchStrategy);
105        }
106
107        TermCriterion termCriterion = new TermCriterion(1);
108
109        AbstractMasterProcess masterProcess =
110                new MasterProcessBasic(pheromoneStructure, ants, termCriterion);
111
112        return new ACOAlgorithm(masterProcess);
113    }
114
115}
