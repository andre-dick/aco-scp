029import eu.andredick.scp.SCPSolution;
030import eu.andredick.scp.SCProblem;
031
032public class AlgorithmConfiguration_StigmergyElite extends AbstractAlgorithmConfiguration {
033
034
035    @Override
036    public void prepareConfigParameters() {
037
038        this.configName = "AlgorithmConfiguration_StigmergyHeuristElite";
039
040        ConfigurationParameter<Float> phInitValue =
041                new ConfigurationParameter<>("pheromonInitValue", 100f);
042        this.addConfigurationParameter(phInitValue);
043        //phInitValue.setExpressions(new Float[]{1f,10f,20f,30f,40f,50f,60f,70f,80f,90f});
044        //phInitValue.setExpressions(new Float[]{50f, 100f, 150f, 200f});
045
046        ConfigurationParameter<Float> evapFactor =
047                new ConfigurationParameter<>("evaporationFactor", 0.5f);
048        this.addConfigurationParameter(evapFactor);
049        evapFactor.setExpressions(new Float[]{0.7f, 0.75f, 0.8f, 0.85f, 0.9f, 0.95f});
050        //evapFactor.setExpressions(new Float[]{0.1f,0.3f,0.5f,0.7f,0.9f});
051
052        ConfigurationParameter<Integer> maxIterations =
053                new ConfigurationParameter<>("maxIterations", 2000);
054        this.addConfigurationParameter(maxIterations);
055
056        this.getParameter("antsize").setCurrentValue(10);
057
058    }
059
060    @Override
061    public AbstractAlgorithm create(SCProblem problem) {
062
063        PheromoneOnSubsets pheromoneStructure =
064                new PheromoneOnSubsets(problem);
065
066        float phInitValue = this.getParameter("pheromonInitValue").getCurrentValue().floatValue();
067        AbstractPheromoneInit pheromoneInitRule =
068                new PheromoneInit(phInitValue);
069
070        float evapFactor = this.getParameter("evaporationFactor").getCurrentValue().floatValue();
071        AbstractPheromoneEvaporation evaporationRule =
072                new PheromoneEvaporation(evapFactor);
073
074        pheromoneStructure.setEvaporationRule(evaporationRule);
075        pheromoneStructure.setPheromoneInitRule(pheromoneInitRule);
076
077        AbstractPheromoneUpdate updateRule =
078                new PheromoneUpdateOnSubsets(pheromoneStructure, new SolutionQualityMin());
079
080        float alpha = 1f;
081        float beta = 0f;
082
083        HeuristicInfoSet heuristicInfoSet = new HeuristicInfoSet();
084
085        AbstractPheromonePerception perceptionRule = new PerceptionSimple();
086
087        AbstractNextStepStrategy nextStepRule =
088                new NextStepStrategyOnSubsetsStochastic(pheromoneStructure, perceptionRule, heuristicInfoSet, new CombinationFactor(alpha, beta));
089
090        AbstractConstructionStrategy constructionStrategy =
091                new ConstructionFromSubsets(nextStepRule);
092
093        AbstractLocalSearchStrategy localSearchStrategy =
094                new LocalSearchStrategyNone();
095
096        int antSize = this.getParameter("antsize").getCurrentValue().intValue();
097        AbstractAnt[] ants = new AbstractAnt[antSize];
098        for (int i = 0; i < ants.length; i++) {
099            ants[i] = new ACOAnt<SCPSolution, SCProblem>(problem, updateRule, constructionStrategy, localSearchStrategy);
100        }
101
102        AbstractTermCriterion termCriterion =
103                new TermCriterionNew(this.getParameter("maxIterations").getCurrentValue().intValue(), 100);
104
105        AbstractMasterProcess masterProcess = new MasterProcessElitistParallel(pheromoneStructure, ants, termCriterion);
106
107        return new ACOAlgorithm(masterProcess);
108    }
109
110}
