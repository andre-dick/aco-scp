001package eu.andredick.configuration;
002
003import eu.andredick.aco.algorithm.ACOAlgorithm;
004import eu.andredick.aco.ant.ACOAnt;
005import eu.andredick.aco.algorithm.AbstractAlgorithm;
006import eu.andredick.aco.ant.AbstractAnt;
007import eu.andredick.aco.combination.CombinationFactor;
008import eu.andredick.aco.construct.AbstractConstructionStrategy;
009import eu.andredick.aco.construct.ConstructionFromSubsets;
010import eu.andredick.aco.heuristic.HeuristicInfoSet;
011import eu.andredick.aco.heuristic.HeuristicRuleBestSubset;
012import eu.andredick.aco.heuristic.HeuristicRuleWeights;
013import eu.andredick.aco.localsearch.AbstractLocalSearchStrategy;
014import eu.andredick.aco.localsearch.LocalSearchStrategyNone;
015import eu.andredick.aco.masterprocess.AbstractMasterProcess;
016import eu.andredick.aco.masterprocess.MasterProcessBasic;
017import eu.andredick.aco.nextstep.AbstractNextStepStrategy;
018import eu.andredick.aco.nextstep.NextStepStrategyOnSubsets;
019import eu.andredick.aco.pheromonassociation.PheromoneOnSubsets;
020import eu.andredick.aco.pheromoneevaporation.AbstractPheromoneEvaporation;
021import eu.andredick.aco.pheromoneevaporation.PheromoneEvaporation;
022import eu.andredick.aco.pheromoneinit.AbstractPheromoneInit;
023import eu.andredick.aco.pheromoneinit.PheromoneInit;
024import eu.andredick.aco.pheromoneperception.AbstractPheromonePerception;
025import eu.andredick.aco.pheromoneperception.PerceptionSimple;
026import eu.andredick.aco.pheromoneupdate.AbstractPheromoneUpdate;
027import eu.andredick.aco.pheromoneupdate.PheromoneUpdateOnSubsets;
028import eu.andredick.aco.solutionquality.SolutionQualityMin;
029import eu.andredick.aco.termination.TermCriterion;
030import eu.andredick.scp.SCProblem;
031
032public class AlgorithmConfiguration_GreedyProbabilistic extends AbstractAlgorithmConfiguration {
033
034
035    @Override
036    public void prepareConfigParameters() {
037
038        this.configName = "AlgorithmConfiguration_Greedy";
039
040        ConfigurationParameter<Float> phInitValue =
041                new ConfigurationParameter<>("pheromonInitValue", 1f);
042        this.addConfigurationParameter(phInitValue);
043
044        ConfigurationParameter<Float> evapFactor =
045                new ConfigurationParameter<>("evaporationFactor", 1f);
046        this.addConfigurationParameter(evapFactor);
047
048        ConfigurationParameter<Float> alpha =
049                new ConfigurationParameter<>("alpha", 0f);
050        this.addConfigurationParameter(alpha);
051
052        ConfigurationParameter<Float> beta =
053                new ConfigurationParameter<>("beta", 1f);
054        this.addConfigurationParameter(beta);
055
056        ConfigurationParameter<Float> q0 =
057                new ConfigurationParameter<>("q0", 0.02f);
058        this.addConfigurationParameter(q0);
059        //q0.setExpressions(new Float[]{0.005f, 0.01f, 0.015f, 0.02f, 0.025f, 0.03f, 0.035f, 0.04f});
060        q0.setExpressions(new Float[]{0.01f, 0.02f, 0.03f, 0.04f});
061
062        ConfigurationParameter<Integer> maxIterations =
063                new ConfigurationParameter<>("maxIterations", 400);
064        this.addConfigurationParameter(maxIterations);
065
066        this.getParameter("antsize").setCurrentValue(10);
067
068    }
069
070    @Override
071    public AbstractAlgorithm create(SCProblem problem) {
072
073        PheromoneOnSubsets pheromoneStructure =
074                new PheromoneOnSubsets(problem.getStructure());
075
076        float phInitValue = this.getParameter("pheromonInitValue").getCurrentValue().floatValue();
077        AbstractPheromoneInit pheromoneInitRule =
078                new PheromoneInit(phInitValue);
079
080        float evapFactor = this.getParameter("evaporationFactor").getCurrentValue().floatValue();
081        AbstractPheromoneEvaporation evaporationRule =
082                new PheromoneEvaporation(evapFactor);
083
084        pheromoneStructure.setEvaporationRule(evaporationRule);
085        pheromoneStructure.setPheromoneInitRule(pheromoneInitRule);
086
087        AbstractPheromoneUpdate updateRule =
088                new PheromoneUpdateOnSubsets(pheromoneStructure, new SolutionQualityMin());
089
090        float alpha = this.getParameter("alpha").getCurrentValue().floatValue();
091        float beta = this.getParameter("beta").getCurrentValue().floatValue();
092        float q0 = this.getParameter("q0").getCurrentValue().floatValue();
093
094        HeuristicInfoSet heuristicInfoSet = new HeuristicInfoSet();
095        heuristicInfoSet.addRule(new HeuristicRuleWeights());
096        heuristicInfoSet.addRule(new HeuristicRuleBestSubset());
097
098        AbstractPheromonePerception perceptionRule = new PerceptionSimple();
099
100        AbstractNextStepStrategy nextStepRule =
101                new NextStepStrategyOnSubsets(
102                        pheromoneStructure, perceptionRule, heuristicInfoSet, new CombinationFactor(alpha, beta), q0);
103
104        AbstractConstructionStrategy constructionStrategy =
105                new ConstructionFromSubsets(nextStepRule);
106
107        AbstractLocalSearchStrategy localSearchStrategy =
108                new LocalSearchStrategyNone();
109
110        int antSize = this.getParameter("antsize").getCurrentValue().intValue();
111
112        AbstractAnt[] ants = new AbstractAnt[antSize];
113        for (int i = 0; i < ants.length; i++) {
114            ants[i] = new ACOAnt(problem, updateRule, constructionStrategy, localSearchStrategy);
115        }
116
117        TermCriterion termCriterion = new TermCriterion(this.getParameter("maxIterations").getCurrentValue().intValue());
118
119        AbstractMasterProcess masterProcess =
120                new MasterProcessBasic(pheromoneStructure, ants, termCriterion);
121
122        return new ACOAlgorithm(masterProcess);
123    }
124
125}
