001package eu.andredick.aco.pheromoneassociation;
002
003import eu.andredick.scp.Structure;
004
005public class PheromoneOnSubsets extends AbstractPheromoneAssociation {
006
007    private float[] pheromoneValues;
008
009    public PheromoneOnSubsets(Structure structure) {
010        super(structure);
011        pheromoneValues = new float[structure.subsetsSize()];
012    }
013
014    @Override
015    public void evaporatePheromones() {
016        for (int i = 0; i < this.pheromoneValues.length; i++) {
017            pheromoneValues[i] = this.evaporationRule.evaporate(pheromoneValues[i]);
018        }
019    }
020
021    @Override
022    public void initPheromones() {
023        for (int i = 0; i < this.pheromoneValues.length; i++) {
024            pheromoneValues[i] = this.pheromoneInitRule.initValue();
025        }
026    }
027
028    @Override
029    public float getPheromone(int i) {
030        return this.pheromoneValues[i];
031    }
032
033
034    @Override
035    public void addPheromone(int i, float ph_delta) {
036        this.pheromoneValues[i] += ph_delta;
037    }
038
039}
