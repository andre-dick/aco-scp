001package eu.andredick.aco.pheromoneassociation;
002
003import eu.andredick.aco.nextstep.AbstractNextStepStrategy;
004import eu.andredick.aco.pheromoneevaporation.AbstractPheromoneEvaporation;
005import eu.andredick.aco.pheromoneinit.AbstractPheromoneInit;
006import eu.andredick.aco.pheromoneperception.AbstractPheromonePerception;
007import eu.andredick.aco.pheromoneupdate.AbstractPheromoneUpdate;
008import eu.andredick.scp.SCProblem;
009
010/**
011 * <b>Ausprägung der Komponente der Pheromonassoziation</b>j, Assoziation mit Paaren von Teilmengen des SCP ({@link SCProblem}).<br>
012 * Kapitel 3.3.1, S. 24, Pheromon-Assoziation<br>
013 * <br>
014 * Bei dieser Ausprägung ist das Pheromon mit Paaren von Teilmengen des SCP (Entität des Problems) assoziiert.<br>
015 * Ein Paar von Teilmengen des SCP kann durch Ameise mit Pheromon markiert werden,<br>
016 * wenn beide Teilmengen in der konstruierten Lösung der Ameise enthalten sind.<br>
017 * <br>
018 * Das mit den Entitäten eines konkreten Problems assoziierte Pheromon kann als ein Layer betrachtet werden,<br>
019 * welchem bestimmte Regeln (Pheromon-Evaporation und Pheromon-Initiierung) zugeordnet sind.<br>
020 * Die Pheromonassoziation ordnet den Entitäten eines konkreten Problems Pheromonkonzentrationen zu.<br>
021 * Daher ist jede Ausprägung der Pheromonassoziation einer bestimmten Ausprägung eines Problems zugewiesen.<br>
022 * Diese Abhängigkeit wird durch die Generische Programmierung {@code <P extends AbstractProblem>} realisiert.<br>
023 * Bei der Ableitung ist damit festzulegen, mit welchem konkreten Problem die Pheromonassoziation besteht.<br>
024 * <br>
025 * Die Pheromonassoziation wird in folgenden Komponenten benötigt:
026 * <ul>
027 * <li>Pheromon-Initiierung {@link AbstractPheromoneInit}</li>
028 * <li>Pheromon-Evaporation {@link AbstractPheromoneEvaporation}</li>
029 * <li>Alternativen-Auswahl {@link AbstractNextStepStrategy}</li>
030 * <li>Pheromon-Markierung  {@link AbstractPheromoneUpdate}</li>
031 * </ul>
032 * <br>
033 * <p><img src="{@docRoot}/images/PheromoneAssociation-a.svg" alt=""></p>
034 * <br>
035 */
036public class PheromoneOnSubsetPairs extends AbstractPheromoneAssociation<SCProblem> {
037
038    /**
039     * Konzentrationen des Pheromons auf Paaren von Teilmengen des SCP, repräsentiert durch ein zweidimensionales symetrisches Array (Matrix).<br>
040     * Die Indizes i, j des Arrays entsprchen den Indizes der zwei Teilmengen des SCP (eindeutige Zuordnung).<br>
041     */
042    private float[][] pheromoneValues;
043
044    /**
045     * Konstruktor
046     *
047     * @param problem Instanz des Set Covering Problem
048     */
049    public PheromoneOnSubsetPairs(SCProblem problem) {
050        super(problem);
051        pheromoneValues = new float[problem.getStructure().subsetsSize()][problem.getStructure().subsetsSize()];
052    }
053
054    /**
055     * Startet einen Zeitschritt der Evaporation des gesamten Pheromons.<br>
056     * Alle Pheromon-Konzentrationen werden mittels der Evaporations-Regel {@link #evaporationRule} angepasst.<br>
057     */
058    @Override
059    public void evaporatePheromones() {
060        for (int i = 0; i < this.pheromoneValues.length; i++) {
061            for (int j = 0; j < this.pheromoneValues[i].length; j++) {
062                pheromoneValues[i][j] = this.evaporationRule.evaporate(pheromoneValues[i][j]);
063            }
064        }
065    }
066
067    /**
068     * Initiiert die Pheromon-Konszentrationen für alle Teilmengen des SCP mittels der Regel für die Initiierung {@link #pheromoneInitRule}.<br>
069     * Diese Methode stellt den Anfangszustand des gesamten Pheromons her, welcher beim Start des ACO-Algorithmus bestehen soll.
070     */
071    @Override
072    public void initPheromones() {
073        for (int i = 0; i < this.pheromoneValues.length; i++) {
074            for (int j = 0; j < this.pheromoneValues[i].length; j++) {
075                pheromoneValues[i][j] = this.pheromoneInitRule.initValue();
076            }
077        }
078    }
079
080    /**
081     * Liefert immer 0.
082     * Diese Methode ist geerbt, kann jedoch nicht verwendet werden, weil nur ein Parameter übergeben werden kann.
083     * Benötigt werden zwei Parameter für die beiden Indizes i,j
084     * @param j Index der Entität des Problems
085     * @return 0, unabhängig von der Eingabe
086     */
087    @Override
088    public float getPheromone(int j) {
089        System.out.println("PheromoneOnSubsetPairs.getPheromone / nicht erlaubt!");
090        return 0;
091    }
092
093    /**
094     * Liefert die mit dem Teilmengen-Paar (i,j) des SCP-Problems assoziierte Pheromon-Konzentration
095     * @param i Index der ersten Teilmenge des Paares
096     * @param j Index der zweiten Teilmenge des Paares
097     * @return Pheromon-Konzentration, die mit dem Teilmengen-Paar (i,j) assoziiert ist
098     */
099    public float getPheromone(int i, int j) {
100        return this.pheromoneValues[i][j];
101    }
102
103    /**
104     * Diese Methode ist geerbt, kann jedoch nicht verwendet werden, weil der Index nur einer Teilmenge als Eingabe übergeben wird.<br>
105     * Keine Änderung des Zustandes der Pheromon-Assoziation unabhängig von der Eingabe.
106     * @param j        Index der Entität des Problems.
107     * @param ph_delta Zugabe der Pheromon-Konzentration
108     */
109    @Override
110    public void addPheromone(int j, float ph_delta) {
111        System.out.println("PheromoneOnSubsetPairs.addPheromone / nicht erlaubt!");
112    }
113
114
115    /**
116     * Fügt zusätzlichen Pheromon durch Addition zum Bestehenden hinzu.<br>
117     * Diese Methode wird in Ausprägungen der Komponente Pheromon-Markierung ({@link AbstractPheromoneUpdate}) verwendet.<br>
118     * Die assoziierte Pheromon-Konzentration ist durch die Indizes des Teilmengen-Paars (i,j) des SCProblems bestimmt.
119     *
120     * @param i Index der ersten Teilmenge des Paares
121     * @param j Index der zweiten Teilmenge des Paares
122     * @param ph_delta Zugabe der Pheromon-Konzentration
123     */
124    public void addPheromone(int i, int j, float ph_delta) {
125        this.pheromoneValues[i][j] += ph_delta;
126    }
127
128}
