001package eu.andredick.aco.pheromoneinit;
002
003import eu.andredick.aco.pheromoneassociation.AbstractPheromoneAssociation;
004
005/**
006 * <b>Ausprägung der Komponente der Pheromon-Initialisierung</b><br>
007 * Kapitel 3.3.3, S. 29, Initialisierung des Pheromons<br>
008 * <br>
009 * Für alle assoziierten Entitäten eines Problems wird der gleiche Startwert der Pheromon-Konzentration erzeugt.<br>
010 * Der Initialwert des Pheromons stellt einen Parameter der Komponente dar.<br>
011 * <br>
012 * Mit der Initialisierung der Pheromonwerte wird ein Anfangszustand für nachfolgende Iteration des ACO-Algorithmus erstellt.<br>
013 * Wie die Pheromon-Konzentrationen initialisiert werden, soll in den Subklassen dieser Komponente festgelegt werden.<br>
014 * <br>
015 * Die Pheromon-Initialisierung wird durch den Masterprozess ({@link eu.andredick.aco.masterprocess.AbstractMasterProcess}) koordiniert.<br>
016 * Dies geschieht, indem die Methode {@link AbstractPheromoneAssociation#initPheromones()} der Komponenten {@link AbstractPheromoneAssociation} aufgerufen wird.<br>
017 * Anschließend benutzt {@link AbstractPheromoneAssociation} eine Ausprägung der Komponente Pheromon-Initialisierung zur Anpassung aller Pheromon-Konzentrationen.<br>
018 * <br>
019 * <p><img src="{@docRoot}/images/PheromoneInit.svg" alt=""></p>
020 */
021public class PheromoneInit implements AbstractPheromoneInit {
022
023    /**
024     * Initial-Wert der Pheromon-Konzentration als Parameter der Komponente<br>
025     */
026    private float pheromoneInitValue;
027
028    /**
029     * Konstruktor
030     *
031     * @param pheromoneInitValue Initial-Wert der Pheromon-Konzentration als Parameter der Komponente
032     */
033    public PheromoneInit(float pheromoneInitValue) {
034        this.pheromoneInitValue = pheromoneInitValue;
035    }
036
037
038    /**
039     * Liefert den Initial-Wert der Pheromon-Konzentration
040     *
041     * @return Initial-Wert der Pheromon-Konzentration
042     */
043    @Override
044    public float initValue() {
045        return this.pheromoneInitValue;
046    }
047}
