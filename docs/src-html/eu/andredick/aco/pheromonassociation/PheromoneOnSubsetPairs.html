001package eu.andredick.aco.pheromonassociation;
002
003import eu.andredick.scp.Structure;
004
005public class PheromoneOnSubsetPairs extends AbstractPheromone {
006
007    private float[][] pheromoneValues;
008
009    public PheromoneOnSubsetPairs(Structure structure) {
010        super(structure);
011        pheromoneValues = new float[structure.subsetsSize()][structure.subsetsSize()];
012    }
013
014    @Override
015    public void evaporatePheromones() {
016        for (int i = 0; i < this.pheromoneValues.length; i++) {
017            for (int j = 0; j < this.pheromoneValues[i].length; j++) {
018                pheromoneValues[i][j] = this.evaporationRule.evaporate(pheromoneValues[i][j]);
019            }
020        }
021    }
022
023    @Override
024    public void initPheromones() {
025        for (int i = 0; i < this.pheromoneValues.length; i++) {
026            for (int j = 0; j < this.pheromoneValues[i].length; j++) {
027                pheromoneValues[i][j] = this.pheromoneInitRule.initValue();
028            }
029        }
030    }
031
032    @Override
033    public float getPheromone(int i) {
034        System.out.println("PheromoneOnSubsetPairs.getPheromone / nicht erlaubt!");
035        return 0;
036    }
037
038
039    public float getPheromone(int i, int j) {
040        return this.pheromoneValues[i][j];
041    }
042
043    @Override
044    public void addPheromone(int i, float ph_delta) {
045        System.out.println("PheromoneOnSubsetPairs.addPheromone / nicht erlaubt!");
046    }
047
048
049    public void addPheromone(int i, int j, float ph_delta) {
050        this.pheromoneValues[i][j] += ph_delta;
051    }
052
053}
