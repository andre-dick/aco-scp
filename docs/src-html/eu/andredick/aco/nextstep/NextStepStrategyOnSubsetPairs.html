001package eu.andredick.aco.nextstep;
002
003import eu.andredick.aco.combination.CombinationRule;
004import eu.andredick.aco.heuristic.HeuristicInfoSet;
005import eu.andredick.aco.pheromoneassociation.PheromoneOnSubsetPairs;
006import eu.andredick.aco.pheromoneperception.AbstractPheromonePerception;
007import eu.andredick.scp.SCPSolution;
008
009import java.util.List;
010import java.util.concurrent.ThreadLocalRandom;
011
012/**
013 * <b>Stochastische / probabilistische Alternativen-Auswahl</b> - Ausprägung der Komponente der Alternativenauswahl, Pheromonassoziation mit Teilmengenpaaren<br>
014 * Kapitel 3.3.5, S. 32, Alternativenauswahl<br>
015 * <br>
016 * Die Stochastische Alternativenauswahl bestimmt aus gegebener Alternativen-Menge eine Auswahl mittels einer Zufallszahl {@code 0 <= z <= 1}.
017 * Das Interval {@code [0.0, 1.0]} wird dazu in genau so viele Bereiche unterteilt, wie es Alternativen gibt.
018 * Die Größe jedes Bereiches wird durch den Wert der zugehörigen Alternative bestimmt.<br>
019 * Die Komponente besitzt keine Parameter.<br>
020 * <br>
021 * Die Komponente Alternativenauswahl wird von Konstruktionsheuristik {@link eu.andredick.aco.construct.AbstractConstructionStrategy}
022 * verwendet, um aus der Menge gegebener Alternativen (Lösungskomponenten) eine Alternative auszuwählen.<br>
023 * Die Auswahl der Alternative stützt sich auf den heuristischen Informationen {@link HeuristicInfoSet} und
024 * den wahrgenommenen Pheromonkonzentrationen {@link AbstractPheromonePerception}, die den Alternativen zugeordnet sind oder für diese berechnet werden.<br>
025 * Mittels der Kombinationsfunktion {@link CombinationRule} wird aus heuristischen Informationen und der wahrgenommenen Pheromonkonzentration ein Wert der Alternative gebildet.<br>
026 * <p><img src="{@docRoot}/images/Nextstep.svg" alt=""></p>
027 */
028public class NextStepStrategyOnSubsetPairs extends
029        AbstractNextStepStrategy<PheromoneOnSubsetPairs, SCPSolution> {
030
031    /**
032     * Konsturktor
033     *
034     * @param pheromonesStructure Pheromonassoziation mit Teilmengenpaaren
035     * @param perceptionRule      Pheromon-Wahrnehmung
036     * @param heuristics          heuristische Informationen
037     * @param combinationRule     Kombinationsfunktion
038     */
039    public NextStepStrategyOnSubsetPairs(PheromoneOnSubsetPairs pheromonesStructure,
040                                         AbstractPheromonePerception perceptionRule,
041                                         HeuristicInfoSet heuristics,
042                                         CombinationRule combinationRule) {
043
044        super(pheromonesStructure, perceptionRule, heuristics, combinationRule);
045    }
046
047    /**
048     * Die Stochastische Alternativenauswahl bestimmt aus gegebener Alternativen-Menge eine Auswahl mittels einer Zufallszahl {@code 0 <= z <= 1}.
049     * Das Interval {@code [0.0, 1.0]} wird dazu in genau so viele Bereiche unterteilt, wie es Alternativen gibt.
050     * Die Größe jedes Bereiches wird durch den Wert der zugehörigen Alternative bestimmt.<br>
051     * <br>
052     * Bei der Assoziation des Pheromons mit Teilmengenpaaren (siehe {@link PheromoneOnSubsetPairs})
053     *
054     * @param solution         partiale Lösung der Ameise
055     * @param availableSubsets verfügbare Alternativen
056     * @return Ergebnis der Auswahl
057     */
058    @Override
059    public Integer chooseSubset(SCPSolution solution, List<Integer> availableSubsets) {
060
061        // Wenn
062        List<Integer> varsList = solution.getSubsets();
063        boolean firstSubsetInSolution = varsList.isEmpty();
064        Integer lastSubset = firstSubsetInSolution ? null : solution.getSubsets().get(solution.getSubsets().size() - 1);
065
066        // Ein Glücksrad mit unterschiedlich großen Abschnitten (im übertagenen Sinne)
067        float[] summands = new float[availableSubsets.size()];
068        float sumSummands = 0f;
069        for (int k = 0; k < summands.length; k++) {
070
071            int subset_j = availableSubsets.get(k);
072            int subset_i = firstSubsetInSolution ? subset_j : lastSubset;
073
074            float ph_k = this.perceptionRule.getPerceptionValue(this.pheromoneStructure.getPheromone(subset_i, subset_j));
075            float hi_k = this.heuristics.getValue(solution, availableSubsets, subset_j);
076            float summand = this.combinationRule.combine(ph_k, hi_k);
077            sumSummands += summand;
078            summands[k] = summand;
079        }
080
081        // Alle Alternativen-Werte = null => zufällige Auswahl
082        if (sumSummands == 0f)
083            return availableSubsets.get(ThreadLocalRandom.current().nextInt(availableSubsets.size()));
084
085        // Glücksrad wird gedreht
086        float z = ThreadLocalRandom.current().nextFloat() * sumSummands;
087        float sumCounter = 0f;
088        for (int k = 0; k < summands.length; k++) {
089            sumCounter += summands[k];
090            if (z < sumCounter) {
091                return availableSubsets.get(k);
092            }
093        }
094        return null;
095    }
096
097}
