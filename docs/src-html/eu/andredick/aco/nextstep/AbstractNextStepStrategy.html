001package eu.andredick.aco.nextstep;
002
003import eu.andredick.aco.combination.CombinationRule;
004import eu.andredick.aco.heuristic.HeuristicInfoSet;
005import eu.andredick.aco.pheromoneassociation.AbstractPheromoneAssociation;
006import eu.andredick.aco.pheromoneperception.AbstractPheromonePerception;
007import eu.andredick.aco.problem.AbstractSolution;
008
009import java.util.List;
010
011/**
012 * <b>Abstrakte Komponente der Alternativen-Auswahl</b><br>
013 * Kapitel 3.3.5, S. 32, Alternativenauswahl<br>
014 * <br>
015 * Die Komponente Alternativenauswahl wird von Konstruktionsheuristik {@link eu.andredick.aco.construct.AbstractConstructionStrategy} verwendet,<br>
016 * um aus der Menge gegebener Alternativen (Lösungskomponenten) eine Alternative auszuwählen.<br>
017 * Die Auswahl der Alternative stützt sich auf den heuristischen Informationen {@link HeuristicInfoSet} und<br>
018 * den wahrgenommenen Pheromonkonzentrationen {@link AbstractPheromonePerception}, die den Alternativen zugeordnet sind oder für diese berechnet werden.<br>
019 * Mittels der Kombinationsfunktion {@link CombinationRule} wird aus heuristischen Informationen und der wahrgenommenen Pheromonkonzentration ein Wert der Alternative gebildet.<br>
020 * <br>
021 * Eine weitere Abhängigkeit besteht von der Komponente Pheromonassoziation (siehe {@link AbstractPheromoneAssociation}.<br>
022 * Um den wahrgenonommenen Pheromonwert bestimmen zu können, wird hierdurch der tatsächliche Pheromonwert ermittelt.
023 * <br>
024 * Realisierungen der abstrakten Komponente müssen die Methode {@link #chooseSubset(AbstractSolution, List)} implementieren.
025 * <p><img src="{@docRoot}/images/Nextstep.svg" alt=""></p>
026 */
027public abstract class AbstractNextStepStrategy<E extends AbstractPheromoneAssociation, S extends AbstractSolution> {
028
029    /**
030     * Pheromonassoziation mit dem zu lösenden AbstractProblem
031     */
032    protected E pheromoneStructure;
033    /**
034     * Pheromon-Wahrnehmung
035     */
036    protected AbstractPheromonePerception perceptionRule;
037    /**
038     * Menge Heuristischer Informationen
039     */
040    protected HeuristicInfoSet heuristics;
041    /**
042     * Kombinationsfunktion (kombiniert die Werte der Pheromon-Wahrnehmung und der Heuristischen Informationen)
043     */
044    protected CombinationRule combinationRule;
045
046
047    /**
048     * Konsturktor
049     *
050     * @param pheromoneStructure Pheromonassoziation
051     * @param perceptionRule     Pheromon-Wahrnehmung
052     * @param heuristics         heuristische Informationen
053     * @param combinationRule    Kombinationsfunktion
054     */
055    public AbstractNextStepStrategy(E pheromoneStructure,
056                                    AbstractPheromonePerception perceptionRule,
057                                    HeuristicInfoSet heuristics,
058                                    CombinationRule combinationRule) {
059
060        this.pheromoneStructure = pheromoneStructure;
061        this.perceptionRule = perceptionRule;
062        this.heuristics = heuristics;
063        this.combinationRule = combinationRule;
064    }
065
066    /**
067     * Abstrakte Methode zur Auswahl einer Alternative aus einer gegebenen Menge.<br>
068     * Schnittstelle zur Konstruktionsheuristik ({@link eu.andredick.aco.construct.AbstractConstructionStrategy}).<br>
069     *
070     * @param solution         partiale Lösung der Ameise
071     * @param availableSubsets verfügbare Alternativen
072     * @return gewählte Alternative
073     */
074    public abstract Integer chooseSubset(S solution, List<Integer> availableSubsets);
075}
