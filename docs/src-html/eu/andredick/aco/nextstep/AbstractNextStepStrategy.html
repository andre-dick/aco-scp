001package eu.andredick.aco.nextstep;
002
003import eu.andredick.aco.combination.CombinationRule;
004import eu.andredick.aco.heuristic.HeuristicInfoSet;
005import eu.andredick.aco.pheromoneassociation.AbstractPheromoneAssociation;
006import eu.andredick.aco.pheromoneperception.AbstractPheromonePerception;
007import eu.andredick.scp.Solution;
008
009import java.util.List;
010
011/**
012 * <b>Abstrakte Komponente der Alternativen-Auswahl</b><br>
013 * Kapitel 3.3.5, S. 32, Alternativenauswahl<br>
014 * <br>
015 * Die Komponente Alternativenauswahl wird von Konstruktionsheuristik {@link eu.andredick.aco.construct.AbstractConstructionStrategy} verwendet,<br>
016 * um aus der Menge gegebener Alternativen (Lösungskomponenten) eine Alternative auszuwählen.<br>
017 * Die Auswahl der Alternative stützt sich auf den heuristischen Informationen {@link HeuristicInfoSet} und<br>
018 * den wahrgenommenen Pheromonkonzentrationen {@link AbstractPheromonePerception}, die den Alternativen zugeordnet sind oder für diese berechnet werden.<br>
019 * Mittels der Kombinationsfunktion {@link CombinationRule} wird aus heuristischen Informationen und der wahrgenommenen Pheromonkonzentration ein Wert der Alternative gebildet.<br>
020 * <br>
021 * Realisierungen der abstrakten Komponente müssen die Methode {@link #chooseSubset(Solution, List)} implementieren.
022 * <p><img src="{@docRoot}/images/Nextstep.svg" alt=""></p>
023 */
024public abstract class AbstractNextStepStrategy<E extends AbstractPheromoneAssociation> {
025
026    /**
027     * Pheromonassoziation mit dem zu lösenden Problem
028     */
029    protected E pheromoneStructure;
030    /**
031     * Pheromon-Wahrnehmung
032     */
033    protected AbstractPheromonePerception perceptionRule;
034    /**
035     * Menge Heuristischer Informationen
036     */
037    protected HeuristicInfoSet heuristics;
038    /**
039     * Kombinationsfunktion (kombiniert die Werte der Pheromon-Wahrnehmung und der Heuristischen Informationen)
040     */
041    protected CombinationRule combinationRule;
042
043
044    /**
045     * Konsturktor
046     *
047     * @param pheromoneStructure Pheromonassoziation
048     * @param perceptionRule     Pheromon-Wahrnehmung
049     * @param heuristics         heuristische Informationen
050     * @param combinationRule    Kombinationsfunktion
051     */
052    public AbstractNextStepStrategy(E pheromoneStructure,
053                                    AbstractPheromonePerception perceptionRule,
054                                    HeuristicInfoSet heuristics,
055                                    CombinationRule combinationRule) {
056
057        this.pheromoneStructure = pheromoneStructure;
058        this.perceptionRule = perceptionRule;
059        this.heuristics = heuristics;
060        this.combinationRule = combinationRule;
061    }
062
063    /**
064     * Abstrakte Methode zur Auswahl einer Alternative aus einer gegebenen Menge.<br>
065     * Schnittstelle zur Konstruktionsheuristik ({@link eu.andredick.aco.construct.AbstractConstructionStrategy}).<br>
066     *
067     * @param solution         partiale Lösung der Ameise
068     * @param availableSubsets verfügbare Alternativen
069     * @return gewählte Alternative
070     */
071    public abstract Integer chooseSubset(Solution solution, List<Integer> availableSubsets);
072}
