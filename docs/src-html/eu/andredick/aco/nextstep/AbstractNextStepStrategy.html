001package eu.andredick.aco.nextstep;
002
003import eu.andredick.aco.combination.CombinationRule;
004import eu.andredick.aco.heuristic.HeuristicInfoSet;
005import eu.andredick.aco.pheromonassociation.AbstractPheromone;
006import eu.andredick.aco.pheromoneperception.AbstractPheromonePerception;
007import eu.andredick.scp.Solution;
008
009import java.util.List;
010
011/**
012 * Kapitel 3.2.5        Alternativenauswahl
013 * Abstrakte Komponente der Alternativenauswahl
014 * Wird von Konstruktionsheuristik verwendet, um die Alternativenauswahl zu treffen
015 */
016public abstract class  AbstractNextStepStrategy<E extends AbstractPheromone> {
017
018    protected E pheromoneStructure;
019    protected AbstractPheromonePerception perceptionRule;
020    protected HeuristicInfoSet heuristics;
021    protected CombinationRule combinationRule;
022
023
024    /**
025     * Konsturktor
026     *
027     * @param pheromoneStructure Pheromonassoziation
028     * @param perceptionRule     Pheromon-Wahrnehmung
029     * @param heuristics         heuristische Informationen
030     * @param combinationRule    Kombinationsfunktion
031     */
032    public AbstractNextStepStrategy(E pheromoneStructure,
033                                    AbstractPheromonePerception perceptionRule,
034                                    HeuristicInfoSet heuristics,
035                                    CombinationRule combinationRule) {
036
037        this.pheromoneStructure = pheromoneStructure;
038        this.perceptionRule = perceptionRule;
039        this.heuristics = heuristics;
040        this.combinationRule = combinationRule;
041    }
042
043    /**
044     * Schnittstelle zur Konstruktionsheuristic (AbstractConstructionStrategy)
045     *
046     * @param solution         partiale Lösung der Ameise
047     * @param availableSubsets verfügbare Alternativen
048     * @return gewählte Alternative
049     */
050    public abstract Integer chooseSubset(Solution solution, List<Integer> availableSubsets);
051}
