001package eu.andredick.aco.nextstep;
002
003import eu.andredick.aco.combination.CombinationRule;
004import eu.andredick.aco.heuristic.HeuristicInfoSet;
005import eu.andredick.aco.pheromoneassociation.PheromoneOnSubsets;
006import eu.andredick.aco.pheromoneperception.AbstractPheromonePerception;
007import eu.andredick.scp.Solution;
008
009import java.util.List;
010import java.util.concurrent.ThreadLocalRandom;
011
012/**
013 * <b>Gemischte Alternativen-Auswahl</b> aus stochastischer und deterministischer Alternativenauswahl<br>
014 * Kapitel 3.3.5, S. 32, Alternativenauswahl<br>
015 * <br>
016 * Die Gemischte Alternativenauswahl liefert entweder das Ergebnis der stochastischen
017 * (siehe {@link NextStepStrategyOnSubsetsStochastic}) oder der deterministischen
018 * Alternativenauswahl (siehe {@link NextStepStrategyOnSubsetsDeterministic}).<br>
019 * Dies wird durch die Realisierung einer Zufallszahl {@code 0.0 <= q <= 1.0} bestimmt, welche in Bezug auf einen
020 * festgelegten Parameter {@code q0} entweder unterhalb {@code (q < q0)} oder oberhalb {@code (q0 <= q)} liegt.<br>
021 * Die Komponente besitzt einen Parameter.<br>
022 * <br>
023 * Die Komponente Alternativenauswahl wird von Konstruktionsheuristik {@link eu.andredick.aco.construct.AbstractConstructionStrategy}
024 * verwendet, um aus der Menge gegebener Alternativen (Lösungskomponenten) eine Alternative auszuwählen.<br>
025 * Die Auswahl der Alternative stützt sich auf den heuristischen Informationen {@link HeuristicInfoSet} und
026 * den wahrgenommenen Pheromonkonzentrationen {@link AbstractPheromonePerception}, die den Alternativen zugeordnet sind oder für diese berechnet werden.<br>
027 * Mittels der Kombinationsfunktion {@link CombinationRule} wird aus heuristischen Informationen und der wahrgenommenen Pheromonkonzentration ein Wert der Alternative gebildet.<br>
028 * <p><img src="{@docRoot}/images/Nextstep.svg" alt=""></p>
029 */
030public class NextStepStrategyOnSubsets extends
031        AbstractNextStepStrategy<PheromoneOnSubsets> {
032
033    /**
034     * Deterministische Alternativenauswahl (Referenz auf ein eigenes Objekt)
035     */
036    private NextStepStrategyOnSubsetsDeterministic nextStepDeterministic;
037
038    /**
039     * Stochastische Alternativenauswahl (Referenz auf ein eigenes Objekt)
040     */
041    private NextStepStrategyOnSubsetsStochastic nextStepStochastic;
042
043    /**
044     * Parameter, der den jeweiligen Einfluss der stochastischen und deterministischen Alternativenauswahl bestimmt
045     */
046    private float q0_parameter;
047
048    /**
049     * Konsturktor
050     *
051     * @param pheromonesStructure Pheromonassoziation
052     * @param perceptionRule      Pheromon-Wahrnehmung
053     * @param heuristics          heuristische Informationen
054     * @param combinationRule     Kombinationsfunktion
055     * @param q0_parameter        Parameter, der den Einfluss der stochastischen und deterministischen Komponente regelt
056     */
057    public NextStepStrategyOnSubsets(PheromoneOnSubsets pheromonesStructure,
058                                     AbstractPheromonePerception perceptionRule,
059                                     HeuristicInfoSet heuristics,
060                                     CombinationRule combinationRule,
061                                     float q0_parameter) {
062
063        super(pheromonesStructure, perceptionRule, heuristics, combinationRule);
064
065        this.q0_parameter = q0_parameter;
066
067        this.nextStepDeterministic =
068                new NextStepStrategyOnSubsetsDeterministic(pheromonesStructure, perceptionRule, heuristics, combinationRule);
069
070        this.nextStepStochastic =
071                new NextStepStrategyOnSubsetsStochastic(pheromonesStructure, perceptionRule, heuristics, combinationRule);
072    }
073
074    /**
075     * Liefert in abhängigkeit des realisierten Wertes einer Zufallszahl {@code q} und des Parameters {@link #q0_parameter}
076     * die probabilistische oder die deterministische Ergebnisse der Alternativenauswahl.
077     * Zur Bestimmung einer Alternative wird eintweder die Komponente {@link #nextStepStochastic} oder {@link #nextStepDeterministic} verwendet.
078     *
079     * @param solution         partiale Lösung der Ameise
080     * @param availableSubsets verfügbare Alternativen
081     * @return Ergebnis der Auswahl
082     */
083    @Override
084    public Integer chooseSubset(Solution solution, List<Integer> availableSubsets) {
085
086        // Zufallszahl 0.0 <= q <= 1.0 erzeugen
087        float q = ThreadLocalRandom.current().nextFloat();
088
089        /* Je nach Realisierung der Zufallszahl und dem Parameterwert q0
090         * erfolgt die Alternativenauswahl entweder stochastisch oder deterministisch
091         */
092        if (q < q0_parameter) {
093            return this.nextStepStochastic.chooseSubset(solution, availableSubsets);
094        } else {
095            return this.nextStepDeterministic.chooseSubset(solution, availableSubsets);
096        }
097    }
098
099}
