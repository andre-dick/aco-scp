001package eu.andredick.aco.nextstep;
002
003import eu.andredick.aco.combination.CombinationRule;
004import eu.andredick.aco.heuristic.HeuristicInfoSet;
005import eu.andredick.aco.pheromonassociation.PheromoneOnSubsets;
006import eu.andredick.aco.pheromoneperception.AbstractPheromonePerception;
007import eu.andredick.scp.Solution;
008
009import java.util.List;
010import java.util.concurrent.ThreadLocalRandom;
011
012/**
013 * Kapitel 3.2.5        Alternativenauswahl
014 * Probabilistische Auswahl - Ausprägung der Komponente der Alternativenauswahl
015 * Wird von Konstruktionsheuristik verwendet, um die Alternativenauswahl zu treffen
016 */
017public class NextStepStrategyOnSubsetsStochastic extends
018        AbstractNextStepStrategy<PheromoneOnSubsets> {
019
020    public NextStepStrategyOnSubsetsStochastic(PheromoneOnSubsets pheromonesStructure,
021                                               AbstractPheromonePerception perceptionRule,
022                                               HeuristicInfoSet heuristics,
023                                               CombinationRule combinationRule) {
024
025        super(pheromonesStructure, perceptionRule, heuristics, combinationRule);
026    }
027
028    /**
029     * Bestimmt probabilistisch die Auswahl, liefert die Alternative mit dem maximalen Wert
030     *
031     * @param solution         partiale Lösung der Ameise
032     * @param availableSubsets verfügbare Alternativen
033     * @return Ergebnis der Auswahl
034     */
035    @Override
036    public Integer chooseSubset(Solution solution, List<Integer> availableSubsets) {
037
038        // Ein Glücksrad mit unterschiedlich großen Abschnitten (im übertagenen Sinne)
039        float[] summands = new float[availableSubsets.size()];
040        float sumSummands = 0f;
041        for (int k = 0; k < summands.length; k++) {
042            int subset = availableSubsets.get(k);
043            float ph_k = this.perceptionRule.getValue(this.pheromoneStructure.getPheromone(subset));
044            float hi_k = this.heuristics.getValue(solution, availableSubsets, subset);
045            float summand = this.combinationRule.combine(ph_k, hi_k);
046            sumSummands += summand;
047            summands[k] = summand;
048        }
049
050        // Alle Alternativen-Werte = null => zufällige Auswahl
051        if (sumSummands == 0f)
052            return availableSubsets.get(ThreadLocalRandom.current().nextInt(availableSubsets.size()));
053
054        // Glücksrad wird gedreht
055        float z = ThreadLocalRandom.current().nextFloat() * sumSummands;
056        float sumCounter = 0f;
057        for (int k = 0; k < summands.length; k++) {
058            sumCounter += summands[k];
059            if (z < sumCounter) {
060                return availableSubsets.get(k);
061            }
062        }
063        return null;
064    }
065
066}
