001package eu.andredick.aco.nextstep;
002
003import eu.andredick.aco.combination.CombinationRule;
004import eu.andredick.aco.heuristic.HeuristicInfoSet;
005import eu.andredick.aco.pheromoneassociation.PheromoneOnSubsets;
006import eu.andredick.aco.pheromoneperception.AbstractPheromonePerception;
007import eu.andredick.scp.Solution;
008
009import java.util.List;
010import java.util.concurrent.ThreadLocalRandom;
011
012/**
013 * <b>Stochastische / probabilistische Alternativenauswahl</b> - Ausprägung der Komponente der Alternativenauswahl<br>
014 * Kapitel 3.3.5, S. 32, Alternativenauswahl<br>
015 * <br>
016 * Die Stochastische Alternativenauswahl bestimmt aus gegebener Alternativen-Menge eine Auswahl mittels einer Zufallszahl {@code 0 <= z <= 1}.
017 * Das Interval {@code [0.0, 1.0]} wird dazu in genau so viele Bereiche unterteilt, wie es Alternativen gibt.
018 * Die Größe jedes Bereiches wird durch den Wert der zugehörigen Alternative bestimmt.
019 * Die Komponente besitzt keine Parameter.<br>
020 * <br>
021 * Die Komponente Alternativenauswahl wird von Konstruktionsheuristik {@link eu.andredick.aco.construct.AbstractConstructionStrategy}
022 * verwendet, um aus der Menge gegebener Alternativen (Lösungskomponenten) eine Alternative auszuwählen.<br>
023 * Die Auswahl der Alternative stützt sich auf den heuristischen Informationen {@link HeuristicInfoSet} und
024 * den wahrgenommenen Pheromonkonzentrationen {@link AbstractPheromonePerception}, die den Alternativen zugeordnet sind oder für diese berechnet werden.<br>
025 * Mittels der Kombinationsfunktion {@link CombinationRule} wird aus heuristischen Informationen und der wahrgenommenen Pheromonkonzentration ein Wert der Alternative gebildet.<br>
026 * <p><img src="{@docRoot}/images/Nextstep.svg" alt=""></p>
027 * */
028public class NextStepStrategyOnSubsetsStochastic extends
029        AbstractNextStepStrategy<PheromoneOnSubsets> {
030
031    public NextStepStrategyOnSubsetsStochastic(PheromoneOnSubsets pheromonesStructure,
032                                               AbstractPheromonePerception perceptionRule,
033                                               HeuristicInfoSet heuristics,
034                                               CombinationRule combinationRule) {
035
036        super(pheromonesStructure, perceptionRule, heuristics, combinationRule);
037    }
038
039    /**
040
041     *
042     * @param solution         partiale Lösung der Ameise
043     * @param availableSubsets verfügbare Alternativen
044     * @return Ergebnis der Auswahl
045     */
046    @Override
047    public Integer chooseSubset(Solution solution, List<Integer> availableSubsets) {
048
049        // Ein Glücksrad mit unterschiedlich großen Abschnitten (im übertagenen Sinne)
050        float[] summands = new float[availableSubsets.size()];
051        float sumSummands = 0f;
052        for (int k = 0; k < summands.length; k++) {
053            int subset = availableSubsets.get(k);
054            float ph_k = this.perceptionRule.getValue(this.pheromoneStructure.getPheromone(subset));
055            float hi_k = this.heuristics.getValue(solution, availableSubsets, subset);
056            float summand = this.combinationRule.combine(ph_k, hi_k);
057            sumSummands += summand;
058            summands[k] = summand;
059        }
060
061        // Alle Alternativen-Werte = null => zufällige Auswahl
062        if (sumSummands == 0f)
063            return availableSubsets.get(ThreadLocalRandom.current().nextInt(availableSubsets.size()));
064
065        // Glücksrad wird gedreht
066        float z = ThreadLocalRandom.current().nextFloat() * sumSummands;
067        float sumCounter = 0f;
068        for (int k = 0; k < summands.length; k++) {
069            sumCounter += summands[k];
070            if (z < sumCounter) {
071                return availableSubsets.get(k);
072            }
073        }
074        return null;
075    }
076
077}
