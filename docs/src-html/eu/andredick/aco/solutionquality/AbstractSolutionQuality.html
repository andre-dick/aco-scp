001package eu.andredick.aco.solutionquality;
002
003import eu.andredick.aco.problem.AbstractSolution;
004import eu.andredick.aco.pheromoneupdate.AbstractPheromoneUpdate;
005
006/**
007 * <b>Abstrakte Komponente der Gütefunktion für Lösungen</b><br>
008 * Kapitel 3.3.12, S. 46, Gütefunktion für Lösungen <br>
009 * <br>
010 * Die Gütefunktion wird von der Komponente Pheromone-Markierung ({@link AbstractPheromoneUpdate}) verwendet.<br>
011 * Sie wird benötigt, um die Menge des Pheromons bei der Markierung von Lösungen vom Zielfunktionswert zu entkoppeln.<br>
012 * Damit kann die Menge des Pheromons beispielsweise auf einen Wertebereich begrenzt werden.<br>
013 * Um die Güte zu bestimmen, ist die Kenntnis über die Beschaffenheit der Lösung ({@link AbstractSolution}) notwendig.<br>
014 * Daher müssen Ausprägungen der abstrakten Gütefunktion festlegen,
015 * für welche Klasse von Lösungen und damit Problemen sie die Güte bestimmen können.<br>
016 *
017 * <p><img src="{@docRoot}/images/SolutionQuality.svg" alt=""></p>
018 *
019 * @param <S> Ausprägung der Abstrakten Klasse von Lösungen
020 */
021public abstract class AbstractSolutionQuality<S extends AbstractSolution> {
022
023    /**
024     * Liefert die Güte der übergebenen Lösung
025     * @param solution zu bewertende Lösung
026     * @return Güte der Lösung
027     */
028    public abstract float getQuality(S solution);
029
030}
