001package eu.andredick.aco.solutionquality;
002
003import eu.andredick.aco.problem.AbstractSolution;
004import eu.andredick.scp.SCPSolution;
005import eu.andredick.aco.pheromoneupdate.AbstractPheromoneUpdate;
006
007/**
008 * <b>Einfache Gütefunktion für SCP-Lösungen</b><br>
009 * Kapitel 3.3.12, S. 46, Gütefunktion für Lösungen <br>
010 * <br>
011 * Die einfache Gütefunktion ist proportional zum Kehrwert des Zielfunktionswertes.<br>
012 * Um den Wertebereich anzuheben wird der Kehrwert des Zielfunktionswertes mit der Anzahl der Teilmengen multipliziert.<br>
013 * <br>
014 * Die Gütefunktion wird von der Komponente Pheromone-Markierung ({@link AbstractPheromoneUpdate}) verwendet.<br>
015 * Sie wird benötigt, um die Menge des Pheromons bei der Markierung von Lösungen vom Zielfunktionswert zu entkoppeln.<br>
016 * Damit kann die Menge des Pheromons beispielsweise auf einen Wertebereich begrenzt werden.<br>
017 * Um die Güte zu bestimmen, ist die Kenntnis über die Beschaffenheit der Lösung ({@link AbstractSolution}) notwendig.<br>
018 * Daher müssen Ausprägungen der abstrakten Gütefunktion festlegen,
019 * für welche Klasse von Lösungen und damit Problemen sie die Güte bestimmen können.<br>
020 *
021 * <p><img src="{@docRoot}/images/SolutionQuality.svg" alt=""></p>
022 */
023public class SolutionQualityMin extends AbstractSolutionQuality<SCPSolution> {
024
025    /**
026     * Liefert die Güte Q der übergebenen Lösung.<br>
027     * Q = n / z<br>
028     * mit n = Anzahl der Teilmengen, z = Zielfunktionswert der Lösung<br>
029     * @param solution: Lösung des SCP
030     * @return Güte der Lösung
031     */
032    @Override
033    public float getQuality(SCPSolution solution) {
034
035        return (float) solution.getProblem().getStructure().subsetsSize() / solution.getValue();
036
037    }
038}
