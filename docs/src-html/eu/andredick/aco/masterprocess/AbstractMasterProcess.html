001package eu.andredick.aco.masterprocess;
002
003import eu.andredick.aco.ant.AbstractAnt;
004import eu.andredick.aco.algorithm.Statistics;
005import eu.andredick.aco.pheromonassociation.AbstractPheromone;
006import eu.andredick.aco.termination.AbstractTermCriterion;
007
008/**
009 * Kapitel 3.2.2        Masterprozess
010 * Abstrakte Komponente des Masterprozess
011 */
012public abstract class AbstractMasterProcess {
013
014    // Die zu koordinierenden Ameisen
015    protected AbstractAnt[] ants;
016
017    // Pheromonassotiation
018    protected AbstractPheromone pheromoneStructure;
019
020    // Abbruchkriterium
021    protected AbstractTermCriterion termCriterion;
022
023    // Statistiken
024    protected Statistics statistics;
025
026    // Konstruktor
027    public AbstractMasterProcess(AbstractPheromone pheromoneStructure, AbstractAnt[] ants, AbstractTermCriterion termCriterion) {
028        this.pheromoneStructure = pheromoneStructure;
029        this.ants = ants;
030        this.termCriterion = termCriterion;
031        this.statistics = new Statistics();
032    }
033
034
035    /**
036     * Schnittstelle zur Klasse ACOAlgorithm
037     * Durch die Methode erfolgt der Start des Masterprozesses
038     */
039    public abstract void start();
040
041    // Liefert Statistiken zur√ºck
042    public Statistics getStatistics() {
043        return statistics;
044    }
045}
