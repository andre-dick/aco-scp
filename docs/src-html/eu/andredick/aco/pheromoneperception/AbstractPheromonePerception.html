001package eu.andredick.aco.pheromoneperception;
002
003/**
004 * <b>Abstrakte Komponente der Pheromon-Wahrnehmung</b><br>
005 * Kapitel 3.3.6, S. 33, Pheromon-Wahrnehmung<br>
006 * <br>
007 * Die Pheromon-Wahrnehmung ist wird bei der Lösungskonstrukton der Ameisen verwendet.<br>
008 * Bei der Lösungskonstruktion findet iterativ die Alternativen-Auswahl der Lösungskomponenten statt.<br>
009 * Bei der Alternativen-Auswahl bezieht die Ameise neben den Heuristischen Informationen auch  das mit den Alternativen assoziierte Pheromon in ihre Entscheidung ein.<br>
010 * Anstatt der Werte der Pheromon-Konzentrationen direkt zu verarbeiten, werden "subjektiv" wahrgenommene Werte verwendet.<br>
011 * Die Pheromon-Wahrnehmung setzt damit für die Alternativen-Auswahl die tatsächliche Pheromon-Konzentration in wahrgenomene Werte um.<br>
012 * <br>
013 * <p><img src="{@docRoot}/images/PheromonePerception.svg" alt=""></p>
014 */
015public abstract class AbstractPheromonePerception {
016
017    /**
018     * Liefert den durch die Ameise wahrgenommenen Wert des Pheromons,<br>
019     * welcher aus der tatsächlichen Pheromon-Konzentration ermittelt wird.<br>
020     * Die tatsächliche Pheromon-Konzentration geht als Eingabe der Methode in Berechnung ein.<br>
021     *
022     * @param pheromoneConcentration tatsächliche Konzentration des Pheromons
023     * @return wahrgenommenen Konzentration des Pheromons
024     */
025    public abstract float getPerceptionValue(float pheromoneConcentration);
026
027}
