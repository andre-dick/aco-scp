001package eu.andredick.aco.construct;
002
003import eu.andredick.aco.nextstep.AbstractNextStepStrategy;
004import eu.andredick.scp.SCProblem;
005import eu.andredick.scp.Solution;
006import eu.andredick.scp.Structure;
007import eu.andredick.tools.ArrayTools;
008import eu.andredick.tools.Tools;
009
010import java.util.List;
011
012/**
013 * <b>Realisierung der Komponente Konstruktionsheuristik.</b><br>
014 * Kapitel 3.3.4, S. 30, Konstruktionsheuristik<br>
015 * <br>
016 * Die Konstruktion der SCP-Lösung einer Ameise erfolgt erfolgt <b>ausgehend von den Teilmengen.</b><br>
017 * Die Konstruktionsheuristik wird von der Klasse ACOAnt (Ameise) verwendet, um neue Lösungen zu konstruieren.<br>
018 * <p><img src="{@docRoot}/images/Construction.svg" alt=""></p>
019 */
020public class ConstructionFromSubsets extends AbstractConstructionStrategy {
021
022    /**
023     * Konstruktor
024     * @param nextStepRule Beliebige Ausprägung der Komponente Alternativenauswahl (siehe {@link AbstractNextStepStrategy})
025     */
026    public ConstructionFromSubsets(AbstractNextStepStrategy nextStepRule) {
027        super(nextStepRule);
028    }
029
030
031    /**
032     * <b>Die Konstruktion der Lösung erfolgt itarativ in folgenden Schritten:</b><br>
033     * 1. Auswahl einer dieser Teilmengen über NextStep und hinzufügen zur Lösung<br>
034     * 2. Hinzufügen dieser Teilmenge zur TabuListe und zur Lösung<br>
035     * 3. bestimmen der Grundelemente, die in der gewählten Teilmenge enthalten sind<br>
036     * 4. Entfernen dieser Grundelemente aus allen Teilmengen, die noch verfügbar sind<br>
037     * 5. entfernen aller Teilmengen, die nach dem letzten Schritt leer geworden sind<br>
038     * 6. entfernen der Grundelemente (3.) aus der Menge noch nicht überdeckter Grundelemente<br>
039     * 7. Zurück zu 1, wenn es weitere nicht überdeckte Grundelemente gibt<br>
040     *
041     * @param problem Das SCProblem, für das eine Lösung zu konstruieren ist
042     * @return konstruierte zulässige Lösung
043     */
044    @Override
045    public Solution construct(SCProblem problem) {
046
047        Structure structure = problem.getStructure();
048        Solution solution = new Solution(problem);
049
050        // Bereitstellen eines Arrays mit Listen von Grundelementen, die in den Teilmengen enthalten sind
051        List<Integer> subsets = Tools.getIndexList(structure.subsetsSize()); //geordnet
052
053        // Bereitstellen der TabuListen für Teilmengen und Grundelemente
054        boolean[] tabuSubsets = ArrayTools.getZeroBoolArray(structure.subsetsSize());
055        boolean[] tabuElements = ArrayTools.getZeroBoolArray(structure.elementsSize());
056
057        // Bereitstellen einer Arbeits-Liste mit Summen von Elementen, die in Teilmengen enthalten sind
058        int[] subsetSizes = new int[structure.subsetsSize()];
059        for (int g = 0; g < subsetSizes.length; g++) {
060            subsetSizes[g] = structure.getElementsInSubset(g).size();
061        }
062
063        // Ein Counter für die Anzahl der noch nicht überdeckter Grundelemente
064        int elementsRemain = structure.elementsSize();
065
066        // Iteration, bis keine Grundelemente mehr zu überdecken sind
067        while (elementsRemain != 0) {
068
069            // Auswahl einer Teilmenge aus Teilmengen, die noch nicht in Lösung sind
070            int subsetIndex = this.nextStepRule.chooseSubset(solution, subsets);
071
072            // Hinzufügen der Teilmenge in die Lösung
073            solution.addSubset(subsetIndex);
074
075            // Markieren der gewählten Teilmenge in der TabuListe
076            tabuSubsets[subsetIndex] = true;
077
078            // bestimmen der Grundelemente, die in der gewählten Teilmenge enthalten sind
079            List<Integer> elementsInSubset = structure.getElementsInSubset(subsetIndex);
080
081            // Im weiteren müssen Teilmengen identifiziert werden, die selbst eine Teilmenge der gewählten Teilmenge sind
082            // d.h. Teilmengen, die einen Teil gleicher Elemente überdecken und sonst keine weiteren
083            for (Integer e : elementsInSubset) {
084                if (!tabuElements[e]) {
085                    List<Integer> subsetsWith_e = structure.getSubsetsWithElement(e);
086                    for (Integer s : subsetsWith_e) {
087                        if (!tabuSubsets[s]) {
088                            subsetSizes[s]--;
089                            if (subsetSizes[s] == 0) {
090                                tabuSubsets[s] = true;
091                                subsets.remove(s);
092                            }
093                        }
094                    }
095                    tabuElements[e] = true;
096                    elementsRemain--;
097                }
098            }
099        }
100
101        return solution;
102    }
103}
