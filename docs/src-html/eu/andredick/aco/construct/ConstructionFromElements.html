001package eu.andredick.aco.construct;
002
003import eu.andredick.aco.nextstep.AbstractNextStepStrategy;
004import eu.andredick.scp.SCProblem;
005import eu.andredick.scp.Solution;
006import eu.andredick.scp.Structure;
007import eu.andredick.tools.Tools;
008
009import java.util.LinkedList;
010import java.util.List;
011import java.util.concurrent.ThreadLocalRandom;
012
013/**
014 * <b>Realisierung der Komponente Konstruktionsheuristik.</b><br>
015 * Kapitel 3.3.4, S. 30, Konstruktionsheuristik<br>
016 * <br>
017 * Die Konstruktion der SCP-Lösung einer Ameise erfolgt erfolgt <b>ausgehend von den Grundelementen.</b><br>
018 * Die Konstruktionsheuristik wird von der Klasse {@link eu.andredick.aco.ant.ACOAnt} (Ameise) verwendet, um neue Lösungen zu konstruieren.<br>
019 * <p><img src="{@docRoot}/images/Construction.svg" alt=""></p>
020 */
021public class ConstructionFromElements extends AbstractConstructionStrategy {
022
023    public ConstructionFromElements(AbstractNextStepStrategy nextStepRule) {
024        super(nextStepRule);
025    }
026
027    /**
028     * <b>Die Konstruktion der Lösung erfolgt itarativ in folgenden Schritten:</b><br>
029     * 1. zufällige Auswahl eines Grundelements, welches noch nicht überdeckt ist<br>
030     * 2. bestimmen der Teilmengen, die das Grundelement enthalten und noch nicht in Lösung sind<br>
031     * 3. Auswahl einer dieser Teilmengen über NextStep<br>
032     * 4. bestimmen der Grundelemente, die in der ausgewählten Teilmenge enthalten sind<br>
033     * 5. hinzufügen der bestimmten Grundelemente zur Menge bereits überdeckter Grundelemente<br>
034     * 6. Zurück zu 1, wenn es weitere nicht überdeckte Grundelemente gibt<br>
035     *
036     * @param problem Das SCProblem, für das eine Lösung zu konstruieren ist
037     * @return konstruierte zulässige Lösung
038     */
039    @Override
040    public Solution construct(SCProblem problem) {
041        Structure structure = problem.getStructure();
042        Solution solution = new Solution(problem);
043
044        List<Integer> elements = Tools.getIndexList(structure.elementsSize()); //geordnet
045        List<Integer> tabuSubsets = new LinkedList<>();
046        //List<Integer> subsets = Tools.getIndexList(structure.subsetsSize()); //geordnet
047
048        while (!elements.isEmpty()) {
049            // Zufällige Wahl eines Elementes aus der Grundmenge
050            int r = ThreadLocalRandom.current().nextInt(elements.size());           //Zufahlszahl 0 .. mE.size()
051            int elementIndex = elements.get(r);
052
053            // Alle Teilmengen für das gewählte Element
054            List<Integer> subsetsWithElement;
055            subsetsWithElement = new LinkedList<>(structure.getSubsetsWithElement(elementIndex));
056
057            // Schnittmenge mit den verbliebenen Teilmengen
058            subsetsWithElement.removeAll(tabuSubsets);
059
060            // Auswahl einer Teilmenge aus den noch verfügbaren Teilmengen
061            int subsetIndex = this.nextStepRule.chooseSubset(solution, subsetsWithElement);
062
063            // Entfernen aller Elemente der gewählten Teilmenge aus den zu berücksichtigenden
064            for (Integer e : structure.getElementsInSubset(subsetIndex)) {
065                elements.remove(e);
066            }
067
068
069            // Entfernen der gewählten Teilmenge aus den zu berücksichtigenden
070            //subsets.remove( (Integer)subsetIndex);
071            tabuSubsets.add((Integer) subsetIndex);
072
073
074            // Einfügen der Teilmenge in die Lösung
075            solution.addSubset(subsetIndex);
076
077        }
078        return solution;
079    }
080}
