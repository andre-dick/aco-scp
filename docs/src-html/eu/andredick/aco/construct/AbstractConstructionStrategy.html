001package eu.andredick.aco.construct;
002
003import eu.andredick.aco.nextstep.AbstractNextStepStrategy;
004import eu.andredick.scp.SCProblem;
005import eu.andredick.scp.Solution;
006
007/**
008 * <b>Abstrakte Konstruktionsheuristik.</b><br>
009 * Kapitel 3.3.4, S. 30, Konstruktionsheuristik<br>
010 * <br>
011 * Die Ausprägungen dieser Komponente werden durch die Klasse {@link eu.andredick.aco.ant.ACOAnt} (Ameise) verwendet, um neue Lösungen zu konstruieren.<br>
012 * Die Konstruktionsheuristik und die Alternativenauswahl hängen logisch stark voneinander ab, <br>
013 * so dass generell nicht jede beliebige Ausprägung der Alternativenauswahl mit einer bestimmten Konstruktionsheuristik verwendet werden kann.<br>
014 * Aus diesem Grund muss bei der Ableitung dieser abstrakten Klasse die generische Alternativenauswahl festgelet werden.<br>
015 * <p><img src="{@docRoot}/images/Construction.svg" alt=""></p>
016 *
017 * @param <E> Ausprägung der Alternativenauswahl, die bei der Konstruktion benutzt werden soll.
018 */
019public abstract class AbstractConstructionStrategy<E extends AbstractNextStepStrategy<?>> {
020
021    /**
022     * Regel für die Alternativenauswahl, die bei der Konstruktion benutzt werden soll.
023     */
024    protected E nextStepRule;
025
026    /**
027     * Konstruktor
028     *
029     * @param nextStepRule Regel für die Alternativenauswahl, die bei der Konstruktion benutzt werden soll.
030     */
031    public AbstractConstructionStrategy(E nextStepRule) {
032        this.nextStepRule = nextStepRule;
033    }
034
035    /**
036     * Die Methode startet den Konstruktionsvorgang und liefert nach seinem Durchlauf eine vollständige Lösung des übergebenen Problems.<br>
037     * Die Methode wird in der Klasse ACOAnt aufgerufen.
038     *
039     * @param problem Das Problem, für welches durch die Ameise eine Lösung konstruiert werden soll.
040     * @return Vollständige konstruierte Lösung
041     */
042    public abstract Solution construct(SCProblem problem);
043}
