001package eu.andredick.aco.pheromoneupdate;
002
003import eu.andredick.aco.pheromonassociation.PheromoneOnSubsetPairs;
004import eu.andredick.aco.solutionquality.AbstractSolutionQuality;
005import eu.andredick.scp.Solution;
006
007import java.util.List;
008
009public class PheromoneUpdateOnSubsetPairs extends
010        AbstractPheromoneUpdate<PheromoneOnSubsetPairs> {
011
012    public PheromoneUpdateOnSubsetPairs(PheromoneOnSubsetPairs pheromoneStructure,
013                                            AbstractSolutionQuality solutionQuality) {
014        super(pheromoneStructure, solutionQuality);
015    }
016
017    @Override
018    public void update(Solution solution) {
019        float delta = this.solutionQuality.getQuality(solution);
020        List<Integer> varsList = solution.getSubsets();
021        int varListSize = varsList.size();
022        for (int i = 0; i < varListSize; i++) {
023            for (int j = i + 1; j < varListSize; j++) {
024                pheromoneStructure.addPheromone(varsList.get(i), varsList.get(j), delta);
025                pheromoneStructure.addPheromone(varsList.get(j), varsList.get(i), delta);
026            }
027        }
028        pheromoneStructure.addPheromone(varsList.get(0), varsList.get(0), delta);
029    }
030}
