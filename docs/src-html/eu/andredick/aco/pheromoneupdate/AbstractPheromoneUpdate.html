001package eu.andredick.aco.pheromoneupdate;
002
003import eu.andredick.aco.pheromoneassociation.AbstractPheromoneAssociation;
004import eu.andredick.aco.solutionquality.AbstractSolutionQuality;
005import eu.andredick.scp.Solution;
006
007/**
008 * Kapitel 3.3.11 <br>
009 * Pheromon-Markierung auf den assoziierten Entitäten des Problems, die in der zu markierenden Lösung enthalten sind. <br>
010 * @param <E>: Ausprägung der Pheromon-Assoziation
011 */
012public abstract class AbstractPheromoneUpdate<E extends AbstractPheromoneAssociation> {
013
014    /**
015     * Pheromon-Assoziationen als generische Refferenz
016     */
017    protected E pheromoneStructure;
018    /**
019     * Gütefunktion der Lösung
020     */
021    protected AbstractSolutionQuality solutionQuality;
022
023    /**
024     * Konstruktor
025     * @param pheromoneStructure: Pheromon-Assoziationen als generische Refferenz.
026     * @param solutionQuality: Gütefunktion der Lösung.q
027     */
028    public AbstractPheromoneUpdate(E pheromoneStructure, AbstractSolutionQuality solutionQuality) {
029        this.pheromoneStructure = pheromoneStructure;
030        this.solutionQuality = solutionQuality;
031    }
032
033    /**
034     * Markiert in Abhängigkeit der Lösung die assoziirten Entitäten des Problems mit Pheromon.
035     * @param solution: Lösung bzw. Ameisen-Pfad, die/der mit Pheromon markiert werden soll.
036     */
037    public abstract void update(Solution solution);
038}
