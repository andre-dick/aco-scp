001package eu.andredick.aco.pheromoneevaporation;
002
003import eu.andredick.aco.pheromoneassociation.AbstractPheromoneAssociation;
004
005/**
006 * <b>Abstrakte Komponente der Pheromon-Evaporation</b><br>
007 * Kapitel 3.3.10, S. 45, Evaporation des Pheromons<br>
008 * <br>
009 * Durch diese Komponente wird der Prozess der stetigen Verdampfung des Pheromons nachgebildet.<br>
010 * Sie stellt einen rekursiven Zusammenhang zwischen den Konzentrationen des Pheromons zweier benachbarter Iterationsschritte her.<br>
011 * <br>
012 * Die Pheromon-Evaporation wird durch den Masterprozess ({@link eu.andredick.aco.masterprocess.AbstractMasterProcess}) koordiniert.<br>
013 * Dies geschieht, indem die Methode {@link AbstractPheromoneAssociation#evaporatePheromones()} der Komponenten {@link AbstractPheromoneAssociation} aufgerufen wird.<br>
014 * Anschließend benutzt {@link AbstractPheromoneAssociation} eine Ausprägung der Komponente Pheromon-Evaporation zur Anpassung aller Pheromon-Konzentrationen.<br>
015 * <br>
016 * <p><img src="{@docRoot}/images/PheromoneEvaporation.svg" alt=""></p>
017 */
018public interface AbstractPheromoneEvaporation {
019
020    /**
021     * Berechnet die Pheromon-Konzentration des nachfolgenden Zeitschrittes (Iterationsschrittes).<br>
022     *
023     * @param pheromoneValue aktuelle Pheromon-Konzentration (Zeitpunkt t)
024     * @return neue Pheromon-Konzentration (Zeitpunkt t+1)
025     */
026    float evaporate(float pheromoneValue);
027
028}
