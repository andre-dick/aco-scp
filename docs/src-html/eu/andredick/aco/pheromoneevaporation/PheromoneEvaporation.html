001package eu.andredick.aco.pheromoneevaporation;
002
003import eu.andredick.aco.pheromoneassociation.AbstractPheromoneAssociation;
004
005/**
006 * <b>Ausprägung der Komponente der Pheromon-Evaporation</b><br>
007 * Kapitel 3.3.10, S. 45 und S.14, Evaporation des Pheromons<br>
008 * <br>
009 * Die Evaporation ist hier mittels der Funktion einer exponentiellen Abnahme modelliert.<br>
010 * Durch diese Komponente wird der Prozess der stetigen Verdampfung des Pheromons nachgebildet.<br>
011 * Sie stellt einen rekursiven Zusammenhang zwischen den Konzentrationen des Pheromons zweier benachbarter Iterationsschritte her.<br>
012 * <br>
013 * Die Komponente besitzt einen Parameter ({@link #evaporationFactor}).<br>
014 * <br>
015 * Die Pheromon-Evaporation wird durch den Masterprozess ({@link eu.andredick.aco.masterprocess.AbstractMasterProcess}) koordiniert.<br>
016 * Dies geschieht, indem die Methode {@link AbstractPheromoneAssociation#evaporatePheromones()} der Komponenten {@link AbstractPheromoneAssociation} aufgerufen wird.<br>
017 * Anschließend benutzt {@link AbstractPheromoneAssociation} eine Ausprägung der Komponente Pheromon-Evaporation zur Anpassung aller Pheromon-Konzentrationen.<br>
018 * <br>
019 * <p><img src="{@docRoot}/images/PheromoneEvaporation.svg" alt=""></p>
020 */
021public class PheromoneEvaporation implements AbstractPheromoneEvaporation {
022
023    /**
024     * Abnahmefaktor (Evaporations-Faktor), mit welchem die Pheromon-Konzentrationen miltipliziert werden
025     */
026    private float evaporationFactor;
027
028    /**
029     * Konstruktor
030     *
031     * @param evaporationFactor Evaporations-Faktor als Parameter der Komponente
032     */
033    public PheromoneEvaporation(float evaporationFactor) {
034        this.evaporationFactor = evaporationFactor;
035    }
036
037    /**
038     * Berechnet die Pheromon-Konzentration des nachfolgenden Zeitschrittes (Iterationsschrittes),<br>
039     * indem der übergebenen Wert der Pheromon-Konzentration mit dem Abnahmefaktor (Evaporations-Faktor) multipliziert wird.
040     *
041     * @param pheromoneValue aktuelle Pheromon-Konzentration (Zeitpunkt t)
042     * @return neue Pheromon-Konzentration (Zeitpunkt t+1)
043     */
044    @Override
045    public float evaporate(float pheromoneValue) {
046        return pheromoneValue * evaporationFactor;
047    }
048}
