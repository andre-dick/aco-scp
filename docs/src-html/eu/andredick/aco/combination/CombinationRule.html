001package eu.andredick.aco.combination;
002
003import eu.andredick.aco.nextstep.AbstractNextStepStrategy;
004
005/**
006 * <b>Abstrakte Kombinationsfunktion.</b><br>
007 * Kapitel 3.3.8, S. 36, Kombinationsfunktion<br>
008 * <br>
009 * Kombiniert den wahrgenommenen Pheromon mit heuristischen Informationen.<br>
010 * Die Kombinationsfunktion wird durch die Komponente Alternativenauswahl verwendet (siehe {@link AbstractNextStepStrategy}).<br>
011 * <p><img src="{@docRoot}/images/Combination.svg" alt=""></p>
012 */
013public abstract class CombinationRule {
014
015
016    /**
017     * Abstrakte Methode zur Berechnung des kombinierten Wertes aus dem wahrgenommenen Pheromon und dem Wert der heuristischen Informationen.<br>
018     * In den ableitende Klassen wird der funktionale Zusammenhang durch diese Methode implementiert.
019     *
020     * @param pheromone wahrgenommenes Pheromon
021     * @param heurist   Wert der heuristischen Informationen
022     * @return Ergebnis der Kombination
023     */
024    public abstract float combine(float pheromone, float heurist);
025
026}
