001package eu.andredick.aco.combination;
002
003/**
004 * <b>Realisierung der Komponente Kombinationsfunktion mittels Multiplikation.</b><br>
005 * Kapitel 3.3.8, S. 36, Kombinationsfunktion<br>
006 * <br>
007 * Kombiniert den wahrgenommenen Pheromon mit heuristischen Informationen <b>durch Multiplikation</b>.<br>
008 * Beide Faktoren werden durch zugehörige Exponenten alpha und beta relativ zueinander gewichtet.<br>
009 * <p><img src="{@docRoot}/images/Combination.svg" alt=""></p>
010 */
011public class CombinationFactor extends CombinationRule {
012
013    /**
014     * Exponent <b>alpha</b> zur relativen Gewichtung des wahrgenommenen Pheromons
015      */
016    float alpha;
017
018    /**
019     * Exponent <b>beta</b> zur relativen Gewichtung des Wertes der heuristischen Informationen
020     */
021    float beta;
022
023    /** Konstruktor mit übergabe der zwei Parameter
024     *
025     * @param alpha Exponent <b>alpha</b> zur relativen Gewichtung des wahrgenommenen Pheromons
026     * @param beta Exponent <b>beta</b> zur relativen Gewichtung des Wertes der heuristischen Informationen
027     */
028    public CombinationFactor(float alpha, float beta) {
029        this.alpha = alpha;
030        this.beta = beta;
031    }
032
033    /** <b>Kombinationsfunktion mittels Multiplikation</b><br>
034     * <code>Math.pow(pheromone, alpha) * Math.pow(heurist, beta)</code>
035     * @param pheromone wahrgenommenes Pheromon
036     * @param heurist   Wert der heuristischen Informationen
037     * @return Ergebnis der Kombination
038     */
039    @Override
040    public float combine(float pheromone, float heurist) {
041        return (float) (Math.pow(pheromone, alpha) * Math.pow(heurist, beta));
042    }
043}
