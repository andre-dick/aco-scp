001package eu.andredick.aco.combination;

003/**
004 * <b>Realisierung der Komponente Kombinationsfunktion mittels Addition.</b><br>
005 * Kapitel 3.3.8, S. 36, Kombinationsfunktion<br>
006 * <br>
007 * Kombiniert den wahrgenommenen Pheromon mit heuristischen Informationen <b>durch Addition</b>.<br>
008 * Beide Summanden werden durch einen Gewichtungsfaktor <b>gamma</b> relativ zueinander gewichtet.<br>
009 * <p><img src="{@docRoot}/images/Combination.svg" alt=""></p>
010 */
011public class CombinationSum extends CombinationRule {
012
013    /**
014     * Gewichtungsfaktor <b>gamma</b>, der die Summanden (Phermonon und heuristischen Informationen) relativ zueinander gewichtet.
015     */
016    float gamma;
017
018    /** Konstruktor mit Ã¼bergabe des Parameters
019     *
020     * @param gamma Gewichtungsfaktor <b>gamma</b>, der die Summanden (Phermonon und heuristischen Informationen) relativ zueinander gewichtet.
021     */
022    public CombinationSum(float gamma) {
023        this.gamma = gamma;
024    }
025
026    /** <b>Kombinationsfunktion mittels Addition.</b><br>
027     * <code>gamma * pheromone + (1 - gamma) * heurist</code>
028     * @param pheromone wahrgenommenes Pheromon
029     * @param heurist   Wert der heuristischen Informationen
030     * @return Ergebnis der Kombination
031     */
032    @Override
033    public float combine(float pheromone, float heurist) {
034        return gamma * pheromone + (1 - gamma) * heurist;
035    }
036}
