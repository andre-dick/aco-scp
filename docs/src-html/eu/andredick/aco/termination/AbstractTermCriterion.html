001package eu.andredick.aco.termination;
002
003import eu.andredick.aco.algorithm.Statistics;
004
005/**
006 * Abstraktes Abbruchkriterium für die Iteration des ACO.<br>
007 * Subklassen konkretisieren die Bedingungen für den Abbruch.<br>
008 * Eine Instanz der Subklasse wird im Masterprozess benutzt.
009 */
010public abstract class AbstractTermCriterion {
011
012    /**
013     * Prüfen, ob die Abbruchbedingung erfüllt ist
014     * @param iteration Zähler der Iterationen
015     * @param statistics Statistiken zum Verlauf der Iteration
016     * @return Wahr, wenn Abbruchbedingungen NICHT erfüllt
017     */
018    public abstract boolean checkTermination(int iteration, Statistics statistics);
019
020}
