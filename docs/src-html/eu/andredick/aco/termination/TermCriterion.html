001package eu.andredick.aco.termination;
002
003import eu.andredick.aco.algorithm.Statistics;
004
005/**
006 * Abbruchkriterium für die Iteration des ACO.<br>
007 * Subklassen konkretisiert die Bedingungen für den Abbruch.<br>
008 * Eine Instanz der Subklasse wird im Masterprozess benutzt.
009 */
010public class TermCriterion extends AbstractTermCriterion {
011
012    /**
013     * Maximale Anzahl der Itarationen
014     */
015    private int maxIterations;
016
017    /**
018     * Konstruktor
019     * @param maxIterations Maximale Anzahl der Itarationen
020     */
021    public TermCriterion(int maxIterations) {
022        this.maxIterations = maxIterations;
023    }
024
025    /**
026     * Prüfen, ob die Abbruchbedingung erfüllt ist.<br>
027     * Abbruchbedingung ist erfüllt, wenn Anzahl der Iterationen ein Maximum überschreitet.
028     * @param iteration: Zähler der Iterationen
029     * @param statistics: Statistiken zum Verlauf der Iteration
030     * @return Wahr, wenn Anzahl der Iterationen NICHT über Maximum
031     */
032    @Override
033    public boolean checkTermination(int iteration, Statistics statistics) {
034        return iteration < maxIterations;
035    }
036}
