001package eu.andredick.aco.termination;
002
003import eu.andredick.aco.algorithm.Statistics;
004import eu.andredick.tools.ArrayTools;
005
006/**
007 * Abbruchkriterium für die Iteration des ACO.<br>
008 * Subklassen konkretisiert die Bedingungen für den Abbruch.<br>
009 * Eine Instanz der Subklasse wird im Masterprozess benutzt.
010 */
011public class TermCriterionNew extends AbstractTermCriterion {
012
013    /**
014     * Maximale Anzahl der Itarationen.
015     */
016    private int maxIterations;
017
018    /**
019     * Maximale Anzahl der Iterationen, seit dem keine bessere Lösunge gefunden wurde.
020     */
021    private int bestValueAgo;
022
023    /**
024     * Konstruktor
025     *
026     * @param maxIterations Maximale Anzahl der Itarationen
027     * @param bestValueAgo  Maximale Anzahl der Iterationen, seit dem keine bessere Lösunge gefunden wurde.
028     */
029    public TermCriterionNew(int maxIterations, int bestValueAgo) {
030        this.maxIterations = maxIterations;
031        this.bestValueAgo = bestValueAgo;
032    }
033
034    /**
035     * Prüfen, ob die Abbruchbedingung erfüllt ist.<br>
036     * Abbruchbedingung ist erfüllt, wenn Anzahl der Iterationen ein Maximum überschreitet <br>
037     * ODER die maximale Anzahl der Iterationen seit bester Lösunge überschritten wird.
038     *
039     * @param iteration  Zähler der Iterationen
040     * @param statistics Statistiken zum Verlauf der Iteration
041     * @return Wahr, wenn Anzahl der Iterationen NICHT über Maximum
042     */
043    @Override
044    public boolean checkTermination(int iteration, Statistics statistics) {
045        if (iteration == 0) return true;
046        Float globMinVal = statistics.getGlobalMinValue();
047        int iterationnOfBestFoundValue = ArrayTools.getIndexOfMinValue(statistics.getIterationMinValuesArray());
048        return iteration - iterationnOfBestFoundValue < bestValueAgo && iteration < maxIterations;
049    }
050}
