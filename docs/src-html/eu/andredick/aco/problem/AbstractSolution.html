001package eu.andredick.aco.problem;
002
003/**
004 * <b>Abstrakte Lösung zur abstrakten Klasse von Problemen</b>, welche durch die ACO-Metaheuristik zu lösen sind<br>
005 * <br>
006 * Eine Lösung im Sinne einer Instanz ist eindeutig einer Instanz eines Problems zugeordnet ({@link #getProblem()}).<br>
007 * Einem Problem dagegen können mehrere Lösungen zugeordnet werden.
008 * <br>
009 * Auf der Klassen-Ebene besteht zwischen Lösung und Problem besteht eine Abhängigkeit der Lösung von der Klasse des Problems.<br>
010 * Bei Ableitung der Abstrakten Lösung muss im Rahmen der generischen Programmierung festgelegt werden, für welche Klasse von Problemen die verwendet wird.<br>
011 * Im Regelfall muss für jede Problem-Klasse eine zugehörige Klasse der Lösung implementiert werden.<br>
012 *
013 * <p><img src="{@docRoot}/images/Problem.svg" alt=""></p>
014 *
015 * @param <P> zugehörige Ausprägung der Klasse von Abstrakten Problemen - eine Klasse von Speziellen Problemmen
016 */
017public abstract class AbstractSolution<P extends AbstractProblem> {
018
019    /**
020     * Liefert die eindeutig zugeordnete Instanz des Problems
021     *
022     * @return Instanz des Problems
023     */
024    public abstract P getProblem();
025
026    /**
027     * Liefert den Zielfunktionswert der Lösung
028     *
029     * @return Zielfunktionswert der Lösung
030     */
031    public abstract float getValue();
032
033    /**
034     * Erzeugt eine leere Lösung der entsprechenden Subklasse<br>
035     * Die Notwendigkeit dieser Methode ergibt sich,
036     * weil auf der generischen Ebene der Definition der ACO-Metaheuristik keine Abhängigkeiten zu Ausprägungen von Komponenten bestehen soll.
037     *
038      * @return Instanz einer leeren Lösung
039     */
040    public abstract AbstractSolution createNew();
041}
