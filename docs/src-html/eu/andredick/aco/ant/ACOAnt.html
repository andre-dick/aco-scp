001package eu.andredick.aco.ant;
002
003import eu.andredick.aco.construct.AbstractConstructionStrategy;
004import eu.andredick.aco.localsearch.AbstractLocalSearchStrategy;
005import eu.andredick.aco.pheromoneupdate.AbstractPheromoneUpdate;
006import eu.andredick.aco.problem.AbstractProblem;
007import eu.andredick.aco.problem.AbstractSolution;
008
009/**
010 * <b>Realisierung der Komponente Ameise</b> durch die Implementierung der abstrakten Methoden der Klasse {@link AbstractAnt}.<br>
011 * <br>
012 * Stellt Methoden bereit, die von der Komponente MasterProzess (siehe {@link eu.andredick.aco.masterprocess.AbstractMasterProcess}) verwendet werden.<br>
013 * Entwurfsmuster Fassade, in dem Methoden mehrerer Klassen zentralisiert durch eine Schnittstelle angeboten werden.<br>
014 * Die Ameise besitzt darüber hinaus Objektvariablen, die ihren Zustand definieren.<br>
015 * <p><img src="{@docRoot}/images/ACOAnt.svg" alt=""></p>
016 */
017public class ACOAnt<S extends AbstractSolution<P>, P extends AbstractProblem> extends AbstractAnt<S> {
018
019    /**
020     * Das AbstractProblem, auf welchem die Lösung (Pfad) gefunden werden soll.
021     */
022    protected P problem;
023
024    /**
025     * Aktuelle Lösung (Pfad) der Ameise.<br>
026     * Die Lösunge kann unvollständig sein.
027     */
028    protected S solution;
029
030    /**
031     * Beliebige Realisierung der Komponente für Pheromon-Markierung
032     */
033    protected AbstractPheromoneUpdate updateRule;
034
035    /**
036     * Beliebige Realisierung der Komponente für Konsturkionsheuristik von Lösungen
037     */
038    protected AbstractConstructionStrategy constructionStrategy;
039
040    /**
041     * Beliebige Realisierung der Komponente für Lokale Suche
042     */
043    protected AbstractLocalSearchStrategy localSearchStrategy;
044
045    /**
046     * Konstruktor
047     *
048     * @param problem              Das AbstractProblem, auf welchem die Lösung bzw. Pfad gefunden werden soll.
049     * @param updateRule           Beliebige Realisierung der Komponente für Konsturkionsheuristik von Lösungen
050     * @param constructionStrategy Beliebige Realisierung der Komponente für Konsturkionsheuristik von Lösungen
051     * @param localSearchStrategy  Beliebige Realisierung der Komponente für Lokale Suche
052     */
053    public ACOAnt(P problem,
054                  AbstractPheromoneUpdate updateRule,
055                  AbstractConstructionStrategy constructionStrategy,
056                  AbstractLocalSearchStrategy localSearchStrategy) {
057
058        this.problem = problem;
059        this.solution = null;
060        this.updateRule = updateRule;
061        this.constructionStrategy = constructionStrategy;
062        this.localSearchStrategy = localSearchStrategy;
063    }
064
065    /**
066     * Startet die Konstrukton einer Ameisenlösung.<br>
067     * Dazu wird die der Ameise zugewiesene Komponente der Konstruktionsheuristik genutzt.<br>
068     * Die Methode wird von dem Masterprozess aufgerufen.<br>
069     * Die von der Ameise erstellte Lösung soll als Objektvariable vorgehalten werden.
070     *
071     * @see AbstractConstructionStrategy#construct(AbstractProblem)
072     */
073    @Override
074    public void constructSolution() {
075        this.solution = (S) this.constructionStrategy.construct(this.problem);
076    }
077
078    /**
079     * Liefert den Zielfunktionswert der Ameisenlösung.<br>
080     * Die Methode wird von dem Masterprozess aufgerufen.<br>
081     * Der Zustand der Ameise kann eine unvollständige Lösung beinhalten!<br>
082     *
083     * @return Zielfunktionswert der Ameisenlösung
084     */
085    @Override
086    public Float evaluateSolution() {
087        return this.solution.getValue();
088    }
089
090    /**
091     * Startet die Markierung der Ameisen-Lösung auf den Entitäten des Problems.<br>
092     * Dazu wird die der Ameise zugewiesene Komponente der Pheromon-Markierung genutzt.<br>
093     * Die Methode wird von dem Masterprozess aufgerufen.<br>
094     *
095     * @see AbstractPheromoneUpdate#update(AbstractSolution)
096     */
097    @Override
098    public void markPheromone() {
099        this.updateRule.update(this.solution);
100    }
101
102    /**
103     * Startet die Verbesserung der konstruierten Ameisenlösung durch Lokale Suche.<br>
104     * Dazu wird die der Ameise zugewiesene Komponente der Lokalen Suche genutzt. <br>
105     * Die Methode wird von dem Masterprozess aufgerufen. <br>
106     *
107     * @see AbstractLocalSearchStrategy#search(AbstractSolution)
108     */
109    @Override
110    public void localSearch() {
111        this.solution = (S) this.localSearchStrategy.search(this.solution);
112    }
113
114    /**
115     * Erneuert den Zustand der Ameise für die nächste Iteration. <br>
116     * Dazu wird die bisherige Ameisenlösung durch eine leere Lösung ersetzt. <br>
117     * Die Methode wird von dem Masterprozess aufgerufen.
118     */
119    @Override
120    public void resetAnt() {
121        this.solution = (S) this.solution.createNew();
122    }
123
124    // Liefert die Lösung der Ameise
125    @Override
126    public S getSolution() {
127        return this.solution;
128    }
129
130    // Schreibt eine Lösung ins Gedächtnis
131    @Override
132    public void setSolution(S solution) {
133        this.solution = solution;
134    }
135
136
137}
