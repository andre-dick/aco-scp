001package eu.andredick.aco.ant;
002
003import eu.andredick.scp.Solution;
004
005/**
006 * <b>Abstrakte Klasse für eine Ameise.</b><br>
007 * <br>
008 * Definiert primär Schnittstellen, die von der Komponente MasterProzess (siehe {@link eu.andredick.aco.masterprocess.AbstractMasterProcess}) verwendet werden.<br>
009 * Entwurfsmuster Fassade, in dem Methoden mehrerer Klassen zentralisiert durch eine Schnittstelle angeboten werden.<br>
010 * Weil alle Methoden abstrakt sind, dient diese Klassse zur Schnittstellendefinition und kann als Java-Interface realisiert sein.<br>
011 * <p><img src="{@docRoot}/images/ACOAnt.svg" alt=""></p>
012 */
013public abstract class AbstractAnt {
014
015    /**
016     * Startet die Konstrukton einer Ameisenlösung.<br>
017     * Die Methode wird von dem Masterprozess aufgerufen.<br>
018     * Die von der Ameise erstellte Lösung soll als Objektvariable vorgehalten werden.
019     */
020    public abstract void constructSolution();
021
022    /**
023     * Startet die Verbesserung der konstruierten Ameisenlösung durch Lokale Suche.<br>
024     * Die Methode wird von dem Masterprozess aufgerufen.<br>
025     * Die lokal verbesserte Lösung soll als Objektvariable vorgehalten werden.
026     */
027    public abstract void localSearch();
028
029    /**
030     * Liefert den Zielfunktionswert der Ameisenlösung.<br>
031     * Die Methode wird von dem Masterprozess aufgerufen.<br>
032     *
033     * @return Zielfunktionswert der Ameisenlösung
034     */
035    public abstract Float evaluateSolution();
036
037    /**
038     * Startet die Markierung der Ameisen-Lösung auf den Entitäten des Problems.<br>
039     * Die Methode wird von dem Masterprozess aufgerufen.<br>
040     */
041    public abstract void markPheromone();
042
043    /**
044     * Erneuert den Zustand der Ameise für die nächste Iteration.<br>
045     * Die Methode wird von dem Masterprozess aufgerufen.<br>
046     */
047    public abstract void resetAnt();
048
049    /**
050     * Liefert die Lösung einer Ameise.<br>
051     * Je nach Zustand der Ameise kann es sich um die gefundene, verbesserte oder leere Lösung handeln.<br>
052     * Abhängig von der Implementierung kann auch unvollständige Lösung geliefert werden.
053     *
054     * @return Lösung einer Ameise
055     */
056    public abstract Solution getSolution();
057
058    /**
059     * Setzt eine Lösung für die Ameise.<br>
060     * Die Methode dient dazu, die Lösung einer Ameise mit einer neuen zu überschreiben.
061     * @param solution neue Lösung
062     */
063    public abstract void setSolution(Solution solution);
064}
