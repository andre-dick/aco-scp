001package eu.andredick.aco.heuristic;
002
003import eu.andredick.scp.Solution;
004
005import java.util.List;
006
007/**
008 * <b>Ausprägung der Komponente der Heuristischen Informationen</b><br>
009 * Kapitel 3.2.7 Heuristische Information<br>
010 * <br>
011 * Eine Ausprägung der heuristischen Information  <b>H_dyn</b><br>
012 * Die Komponente wird bei der Alternativenauswahl verwendet (siehe {@link eu.andredick.aco.nextstep.AbstractNextStepStrategy}).<br>
013 */
014public class HeuristicRuleBestSubset implements HeuristicRule {
015
016    /**
017     * Berechnet die Anzahl möglicher neuer Überdeckungen der Grundelemente durch die Auswahl der Teilmenge.<br>
018     * Dazu wird für jedes Grundelement, welches durch die Teilmenge <i>subset</i> überdeckt wird, geprüft,<br>
019     * ob dieses bereits durch die Teilmengen aus der partialen Lösung überdeckt ist.<br>
020     * Die Summe aller Grundelemente, für die dies nicht zutrifft, ist der Wert der Heuristischen Information.
021     *
022     * @param solution:         partialle Lösung im Konstruktionsprozess der Ameise
023     * @param availableSubsets alle verfügbaren Teilmengen
024     * @param subset           die zu bewertende Teilmenge
025     * @return Summe der möglichen Neuüberdeckungen
026     */
027    @Override
028    public float getValue(Solution solution, List<Integer> availableSubsets, Integer subset) {
029
030        // Bestimmen der Grundelemente, welche in der Teilmenge "subset" enthalten sind
031        List<Integer> elementsInSubset = solution.getProblem().getStructure().getElementsInSubset(subset);
032
033        // Initiieren des Zählers "sum"
034        int sum = 0;
035
036        // Iteration über alle vorab bestimmten Grundelemente
037        for (Integer e : elementsInSubset) {
038
039            // Wenn das aktuelle Grundelement "e" nicht bereits durch die partiale Lösung überdeckt ist,
040            // dann inkrementiere den Zähler "sum"
041            if (!solution.isElementCovered(e)) sum++;
042        }
043
044        // Rückgabe der Summe der möglichen Neuüberdeckungen
045        return ((float) sum);
046    }
047}
