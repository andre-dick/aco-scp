001package eu.andredick.aco.heuristic;
002
003import eu.andredick.scp.Solution;
004
005import java.util.List;
006
007/**
008 * Kapitel 3.2.7        Heuristische Information
009 * Eine Ausprägung der heuristischen Information Regel H_dyn
010 */
011public class HeuristicRuleBestSubset implements HeuristicRule {
012
013    /**
014     * Berechnet die mögliche Anzahl neuer Überdeckungen der Grundelemente
015     *
016     * @param solution:         partialle Lösung im Konstruktionsprozess der Ameise
017     * @param availableSubsets: alle verfügbaren Alternativen
018     * @param subset:           die zu bewertende Alternative
019     * @return
020     */
021    @Override
022    public float getValue(Solution solution, List<Integer> availableSubsets, Integer subset) {
023
024        List<Integer> elementsInSubset = solution.getProblem().getStructure().getElementsInSubset(subset);
025
026        int sum = 0;
027        for (Integer e : elementsInSubset) {
028            if (!solution.isElementCovered(e)) sum++;
029        }
030        return ((float) sum);
031    }
032}
