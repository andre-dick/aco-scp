001package eu.andredick.aco.heuristic;
002
003import eu.andredick.scp.SCPSolution;
004
005import java.util.List;
006
007/**
008 * <b>Ausprägung der Komponente der Heuristischen Informationen</b><br>
009 * Kapitel 3.3.7, S. 34, Heuristische Information<br>
010 * <br>
011 * Eine Ausprägung der heuristischen Information  <b>H_dyn</b><br>
012 * Die Komponente wird bei der Alternativenauswahl verwendet (siehe {@link eu.andredick.aco.nextstep.AbstractNextStepStrategy}).<br>
013 * <p><img src="{@docRoot}/images/Heuristics.svg" alt=""></p>
014 */
015public class HeuristicRuleBestSubset implements HeuristicRule<SCPSolution> {
016
017    /**
018     * Berechnet die Anzahl möglicher neuer Überdeckungen der Grundelemente durch die Auswahl der Teilmenge.<br>
019     * Dazu wird für jedes Grundelement, welches durch die Teilmenge <i>subset</i> überdeckt wird, geprüft,<br>
020     * ob dieses bereits durch die Teilmengen aus der partialen Lösung überdeckt ist.<br>
021     * Die Summe aller Grundelemente, für die dies nicht zutrifft, ist der Wert der Heuristischen Information.
022     *
023     * @param solution:         partialle Lösung im Konstruktionsprozess der Ameise
024     * @param availableSubsets alle verfügbaren Teilmengen
025     * @param subset           die zu bewertende Teilmenge
026     * @return Summe der möglichen Neuüberdeckungen
027     */
028    @Override
029    public float getValue(SCPSolution solution, List<Integer> availableSubsets, Integer subset) {
030
031        // Bestimmen der Grundelemente, welche in der Teilmenge "subset" enthalten sind
032        List<Integer> elementsInSubset = solution.getProblem().getStructure().getElementsInSubset(subset);
033
034        // Initiieren des Zählers "sum"
035        int sum = 0;
036
037        // Iteration über alle vorab bestimmten Grundelemente
038        for (Integer e : elementsInSubset) {
039
040            // Wenn das aktuelle Grundelement "e" nicht bereits durch die partiale Lösung überdeckt ist,
041            // dann inkrementiere den Zähler "sum"
042            if (!solution.isElementCovered(e)) sum++;
043        }
044
045        // Rückgabe der Summe der möglichen Neuüberdeckungen
046        return ((float) sum);
047    }
048}
