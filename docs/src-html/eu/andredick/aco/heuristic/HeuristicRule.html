001package eu.andredick.aco.heuristic;
002
003import eu.andredick.scp.Solution;
004
005import java.util.List;
006
007/**
008 * <b>Schnittstelle für Klassen der Heuristischen Information</b><br>
009 * Kapitel 3.2.7 Heuristische Information<br>
010 * <br>
011 * Ausprägungen der Komponente der Heuristische Informationen müssen diese Schnittstelle implemenntieren.<br>
012 * Die Komponente wird bei der Alternativenauswahl verwendet (siehe {@link eu.andredick.aco.nextstep.AbstractNextStepStrategy}).<br>
013 * <p><img src="{@docRoot}/images/Heuristics.svg" alt=""></p>
014 */
015public interface HeuristicRule {
016
017    /**
018     * Liefert den Wert der Heuristischen Informationen
019     *
020     * @param solution         partialle Lösung im Konstruktionsprozess der Ameise
021     * @param availableSubsets alle verfügbaren Alternativen
022     * @param subset           Alternative, für die die Heuristischen Informationen geliefert werden sollen
023     * @return Wert der Heuristischen Informationen
024     */
025    public float getValue(Solution solution, List<Integer> availableSubsets, Integer subset);
026
027}
