001package eu.andredick.aco.heuristic;
002
003import eu.andredick.scp.ObjectiveFunction;
004import eu.andredick.scp.Solution;
005
006import java.util.List;
007
008/**
009 * <b>Ausprägung der Komponente der Heuristischen Informationen</b><br>
010 * Kapitel 3.2.7 Heuristische Information<br>
011 * <br>
012 * Eine Ausprägung der heuristischen Information <b>H_stat</b><br>
013 * Die Komponente wird bei der Alternativenauswahl verwendet (siehe {@link eu.andredick.aco.nextstep.AbstractNextStepStrategy}).<br>
014 * <p><img src="{@docRoot}/images/Heuristics.svg" alt=""></p>
015 */
016public class HeuristicRuleWeights implements HeuristicRule {
017
018    /**
019     * Bestimmt die Kosten der Teilmenge <i>subset</i>.<br>
020     * Die Kosten der Teilmengen sind durch den Kostenvektor der Zielfunktion des SCP definiert.
021     * @see ObjectiveFunction#getWeights()
022     *
023     * @param solution         partiale Lösung im Konstruktionsprozess der Ameise
024     * @param availableSubsets alle verfügbaren Teilmengen
025     * @param subset           die zu bewertende Teilmenge
026     * @return Kosten der Teilmenge
027     */
028    @Override
029    public float getValue(Solution solution, List<Integer> availableSubsets, Integer subset) {
030
031        return 1f / solution.getProblem().getObjectiveFunction().getWeights()[subset];
032
033    }
034}
