001package eu.andredick.aco.heuristic;
002
003import eu.andredick.aco.problem.AbstractSolution;
004
005import java.util.ArrayList;
006import java.util.List;
007
008/**
009 * <b>Eine Menge für Heuristische Informationen</b><br>
010 * Kapitel 3.3.7, S. 34, Heuristische Information<br>
011 * <br>
012 * Die Klasse ist ein Behälter für unterschiedliche Heuristische Informationen,<br>
013 * die eine Ameise bei der Lösungskonsturktion verwenden soll.<br>
014 * Die Komponente wird bei der Alternativenauswahl verwendet (siehe {@link eu.andredick.aco.nextstep.AbstractNextStepStrategy}).<br>
015 * <p><img src="{@docRoot}/images/Heuristics.svg" alt=""></p>
016 */
017public class HeuristicInfoSet<S extends AbstractSolution> implements HeuristicRule<S> {
018
019    /**
020     * Menge von Ausprägungen der Komponente Heuristischer Informationen
021     */
022    private ArrayList<HeuristicRule> rules;
023
024    /**
025     * Konstruktor
026     */
027    public HeuristicInfoSet() {
028        rules = new ArrayList<>();
029    }
030
031    /**
032     * Fügt eine Ausprägung von Heuristischen Information hinzu
033     *
034     * @param rule Ausprägung von Heuristischen Information
035     */
036    public void addRule(HeuristicRule rule) {
037        rules.add(rule);
038    }
039
040    /**
041     * Leitet die Anfrage zur Bereichnung der der Werte an alle enthaltenen Heuristiken weiter,<br>
042     * bildet ein Produkt der erhaltenen Werte und liefert diesen zurück.
043     *
044     * @param solution         partiale Lösung im Konstruktionsprozess der Ameise
045     * @param availableSubsets alle verfügbaren Alternativen
046     * @param subset           Alternative, für die die Heuristischen Informationen geliefert werden sollen
047     * @return Wert der Heuristischen Informationen
048     */
049    public float getValue(S solution, List<Integer> availableSubsets, Integer subset) {
050        float product = 1;
051        for (HeuristicRule rule : rules) {
052            product *= rule.getValue(solution, availableSubsets, subset);
053        }
054        return product;
055    }
056}
