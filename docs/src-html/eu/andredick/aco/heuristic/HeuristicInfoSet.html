001package eu.andredick.aco.heuristic;
002
003import eu.andredick.scp.Solution;
004
005import java.util.ArrayList;
006import java.util.List;
007
008/**
009 * <b>Eine Menge für Heuristische Informationen</b><br>
010 * Kapitel 3.2.7 Heuristische Information<br>
011 * <br>
012 * Die Klasse ist ein Behälter für unterschiedliche Heuristische Informationen,<br>
013 * die eine Ameise bei der Lösungskonsturktion verwenden soll.<br>
014 * Die Komponente wird bei der Alternativenauswahl verwendet (siehe {@link eu.andredick.aco.nextstep.AbstractNextStepStrategy}).<br>
015 */
016public class HeuristicInfoSet implements HeuristicRule{
017
018    /**
019     * Menge von Ausprägungen der Komponente Heuristischer Informationen
020     */
021    private ArrayList<HeuristicRule> rules;
022
023    /**
024     * Konstruktor
025     */
026    public HeuristicInfoSet() {
027        rules = new ArrayList<>();
028    }
029
030    /**
031     * Fügt eine Ausprägung von Heuristischen Information hinzu
032     *
033     * @param rule Ausprägung von Heuristischen Information
034     */
035    public void addRule(HeuristicRule rule) {
036        rules.add(rule);
037    }
038
039    /**
040     * Leitet die Anfrage zur Bereichnung der der Werte an alle enthaltenen Heuristiken weiter,<br>
041     * bildet ein Produkt der erhaltenen Werte und liefert diesen zurück.
042     *
043     * @param solution         partiale Lösung im Konstruktionsprozess der Ameise
044     * @param availableSubsets alle verfügbaren Alternativen
045     * @param subset           Alternative, für die die Heuristischen Informationen geliefert werden sollen
046     * @return Wert der Heuristischen Informationen
047     */
048    public float getValue(Solution solution, List<Integer> availableSubsets, Integer subset) {
049        float product = 1;
050        for (HeuristicRule rule : rules) {
051            product *= rule.getValue(solution, availableSubsets, subset);
052        }
053        return product;
054    }
055}
