005/**
006 * <b>Abstrakte Komponente der Lokalen Suche</b><br>
007 * Kapitel 3.3.9, S. 37, Lokale Suche<br>
008 * <br>
009 * Die Lokalen Suche wird durch die Ameise in der Klasse {@link eu.andredick.aco.ant.ACOAnt} verwendet.<br>
010 * Die Lokale Suche erfolgt auf Basis der zuvor konstruierten Lösung und liefert ggf verbesserte Lösung als Ergebnis.

012 */
013public abstract class AbstractLocalSearchStrategy {
014
015    /**
016     * Startet die Lokale Suche anhand der Ausgangslösung und liefert eine verbesserte oder gleiche Lösung zurück.<br>
017     * Schnittstelle zu der Komponente der Ameise der Klasse {@link eu.andredick.aco.ant.ACOAnt}
018     *
019     * @param solution Ausgangslösung
020     * @return Verbesserte Lösung
021     */
022    public abstract Solution search(Solution solution);
023
024}
